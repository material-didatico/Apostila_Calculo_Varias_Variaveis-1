%------------------------------------------------------------------------------%
\begin{question}
  Encontre os valores de \(\D{f}{x}\) e \(\D{f}{y}\) no ponto \((4, -5)\)
  para a função
  \[
    f(x, y) = x^2 + 3xy + y - 1
  \]
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \begin{align*}
    \onslide<+->{\D{f}{x}}
    \onslide<+->{& = \D{}{x} \left(x^2 + 3xy + y - 1\right)} \\
    \onslide<+->{& = \D{x^2}{x} + \D{3xy}{x} + \D{y}{x} - \D{1}{x}} \\
    \onslide<+->{& = 2x + 3y\D{x}{x} + 0 - 0} \\
    \onslide<+->{& = 2x + 3y}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \[
    \D{f}{x}(4, -5)
    \pause
    = \left(2x + 3y\right)\evalat{(x, y)=(4, -5)}{}
    \pause
    = 2\cdot 4 + 3(-5)
    \pause
    = -7
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \begin{align*}
    \onslide<+->{\D{f}{y}}
    \onslide<+->{& = \D{}{y} \left(x^2 + 3xy + y - 1\right)} \\
    \onslide<+->{& = \D{x^2}{y} + \D{3xy}{y} + \D{y}{y} - \D{1}{y}} \\
    \onslide<+->{& = 0 + 3x\D{y}{y} + \D{y}{y} - 0} \\
    \onslide<+->{& = 3x + 1}
    \onslide<+->{\hspace{60mm}\text{Cadê o $y$?}}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \[
    \D{f}{y}(4, -5)
    \pause
    = \left(3x + 1\right)\evalat{(x, y)=(4, -5)}{}
    \pause
    = 3\cdot 4 + 1
    \pause
    = 13
  \]
\end{resolution}
