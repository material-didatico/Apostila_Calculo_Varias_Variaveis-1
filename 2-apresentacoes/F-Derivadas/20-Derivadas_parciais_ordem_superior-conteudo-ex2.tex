%------------------------------------------------------------------------------%
\begin{question}
  Encontre as derivadas de segunda ordem mistas da função
  \(
    f(x, y) = \arctan\left(\frac{y}{x}\right)
  \)

  \pause
  Dica
  \(
    \frac{d}{dt} \arctan(t)
    = \arctan'(t)
    = \frac{1}{1+t^2}
  \)

  \pause
  \vspace{\baselineskip}
  $f$ \emph{não} é contínua no plano

  \pause
  \vspace{\baselineskip}
  Precisamos calcular $f_{xy}$\, e\, $f_{yx}$
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{$f_x$}
\begin{columns}
\column{0.5\linewidth}
  \begin{align*}
    \onslide<+->{f_x(x, y)}
    \onslide<+->{& = \D{f}{x} \\}
    \onslide<+->{& = \D{}{x}\left(\arctan\left(\frac{y}{x}\right)\right) \\}
    \onslide<+->{& = \arctan'\left(\frac{y}{x}\right)\D{}{x}\left(\frac{y}{x}\right) \\}
    \onslide<+->{& = \arctan'\left(\frac{y}{x}\right)y\D{x^{-1}}{x} \\}
    \onslide<+->{& = \arctan'\left(\frac{y}{x}\right)y\left(-x^{-2}\right) \\}
    \onslide<+->{& = \arctan'\left(\frac{y}{x}\right)\frac{-y}{x^2}}
  \end{align*}
\column{0.5\linewidth}
  \onslide<+->{Lembrando que \( \arctan'(t) = \frac{1}{1+t^2} \)}
  \begin{align*}
    \onslide<+->{f_x(x, y)
    & = \frac{1}{1+\left(\dfrac{y}{x}\right)^2}     \;\frac{-y}{x^2} \\}
    \onslide<+->{& = \frac{1}{\dfrac{x^2+y^2}{x^2}} \;\frac{-y}{x^2} \\}
    \onslide<+->{& = \frac{x^2}{x^2+y^2} \;\frac{-y}{x^2} \\}
    \onslide<+->{& = \frac{-y}{x^2+y^2}}
  \end{align*}
\end{columns}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{$f_y$}
\begin{columns}
  \column{0.5\linewidth}
  \begin{align*}
    \onslide<+->{f_y(x, y)}
    \onslide<+->{& = \D{f}{y} \\}
    \onslide<+->{& = \D{}{y}\left(\arctan\left(\frac{y}{x}\right)\right)             \\}
    \onslide<+->{& = \arctan'\left(\frac{y}{x}\right)\D{}{y}\left(\frac{y}{x}\right) \\}
    \onslide<+->{& = \arctan'\left(\frac{y}{x}\right)\frac{1}{x}\D{y}{y}             \\}
    \onslide<+->{& = \arctan'\left(\frac{y}{x}\right)\frac{1}{x}}
  \end{align*}
  \column{0.5\linewidth}
  \onslide<+->{Lembrando que \( \arctan'(t) = \frac{1}{1+t^2} \)}
  \begin{align*}
    \onslide<+->{f_y(x, y)
      & = \frac{1}{1+\left(\dfrac{y}{x}\right)^2}   \;\frac{1}{x} \\}
    \onslide<+->{& = \frac{1}{\dfrac{x^2+y^2}{x^2}} \;\frac{1}{x} \\}
    \onslide<+->{& = \frac{x^2}{x^2+y^2}\;\frac{1}{x}             \\}
    \onslide<+->{& = \frac{x}{x^2+y^2}}
  \end{align*}
\end{columns}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{$f_{xy}$}
\begin{columns}[t]
\column{0.4\linewidth}
    \begin{align*}
    \onslide<+->{f_{xy}(x, y)}
    \onslide<+->{& = \left(f_x\right)_y(x, y) \\}
    \onslide<+->{& = \D{}{y}\left(\D{f}{x}\right) \\}
    \onslide<+->{& = \D{}{y}\left(\frac{-y}{x^2+y^2}\right) \\}
    \onslide<+->{& = -\frac{\displaystyle (x^2+y^2)\D{y}{y}-y\D{}{y}(x^2+y^2)}{(x^2+y^2)^2}}
  \end{align*}
\column{0.6\linewidth}
  \begin{align*}
    \onslide<+->{f_{xy}(x, y)
    & = -\frac{\displaystyle x^2+y^2-y\left(\D{x^2}{y}+\D{y^2}{y}\right)}{(x^2+y^2)^2} \\}
    \onslide<+->{& = -\frac{x^2+y^2-y(2y)}{(x^2+y^2)^2} \\}
    \onslide<+->{& = -\frac{x^2-y^2}{(x^2+y^2)^2} \\}
    \onslide<+->{& = \frac{y^2-x^2}{(x^2+y^2)^2}}
  \end{align*}
\end{columns}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{$f_{yx}$}
\begin{columns}
\column{0.4\linewidth}
  \begin{align*}
    \onslide<+->{f_{yx}(x, y)}
    \onslide<+->{& = \left(f_y\right)_x(x, y) \\}
    \onslide<+->{& = \D{}{x}\left(\D{f}{y}\right) \\}
    \onslide<+->{& = \D{}{x}\left(\frac{x}{x^2+y^2}\right) \\}
    \onslide<+->{& = \frac{\displaystyle \left(x^2+y^2\right)\D{x}{x} - x \D{}{x}\left(x^2+y^2\right)}
             {\left(x^2+y^2\right)^2}}
  \end{align*}
\column{0.6\linewidth}
  \begin{align*}
    \onslide<+->{f_{yx}(x, y)
    & = \frac{\displaystyle x^2+y^2 - x \left(\D{x^2}{x}+\D{y^2}{x}\right)}
             {\left(x^2+y^2\right)^2} \\}
    \onslide<+->{& = \frac{x^2+y^2 - x \left(2x\right)}{\left(x^2+y^2\right)^2} \\}
    \onslide<+->{& = \frac{y^2 - x^2}{\left(x^2+y^2\right)^2}}
  \end{align*}
\end{columns}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Resposta}
  \begin{align*}
    f_{xy}(x, y) & = \frac{y^2-x^2}{\left(x^2+y^2\right)^2} \\[3mm]
    f_{yx}(x, y) & = \frac{y^2-x^2}{\left(x^2+y^2\right)^2}
  \end{align*}
\end{resolution}
