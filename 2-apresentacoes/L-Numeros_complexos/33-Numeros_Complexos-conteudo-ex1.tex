%------------------------------------------------------------------------------%
\begin{question}
  Simplifique as expressões
  \[
    a = i^7
  \]

  \[
    b = i^{57}
  \]

  \[
    c = \frac{i^4 + i^{17}}{i^4}
  \]
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \[
    a
    = i^7
    \pause
    = i^4 \, i^3
    \pause
    = i^3
    \pause
    = i^2 i
    \pause
    = -i
  \]

  \pause
  \[
    b
    = i^{57}
    \pause
    = i^{4\times 14} \, i^1
    \pause
    = i
  \]

  \pause
  \[
    c
    = \frac{i^4 + i^{17}}{i^4}
    \pause
    = \frac{i^4}{i^4} + \frac{i^{17}}{i^4}
    \pause
    = 1 + i^{13}
    \pause
    = 1 + i^{4\times 3} \, i
    \pause
    = 1 + i
  \]
\end{resolution}
