\begin{question}
  Calcule
  \(
    z = \left[2\left(\cos\left(\frac{\pi}{6}\right) + i\sin\left(\frac{\pi}{6}\right)\right)\right]^4
  \)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \[
    u = 2\left(\cos\left(\frac{\pi}{6}\right) + i\sin\left(\frac{\pi}{6}\right)\right)
  \]

  \[
    \pause \rho
    \pause = \abs{u}
    \pause = 2
    \qquad\qquad
    \pause \varphi
    \pause = \arg(u)
    \pause = \frac{\pi}{6}
  \]

  \[
    \pause
    z = u^4
    \pause
    = \rho^4 \big[\cos(4\varphi) + i\sin(4\varphi)\big]
  \]
  \[
    \hphantom{z^4}
    \pause
    = 2^4 \left[\cos\left(4\frac{\pi}{6}\right) + i\sin\left(4\frac{\pi}{6}\right)\right]
    \pause
    = 16  \left[\cos\left(\frac{2\pi}{3}\right) + i\sin\left(\frac{2\pi}{3}\right)\right]
  \]
\end{resolution}
