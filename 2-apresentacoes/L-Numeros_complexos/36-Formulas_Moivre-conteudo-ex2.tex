%------------------------------------------------------------------------------%
\begin{question}
  Calcule
  \(
    z = \left(1+\sqrt{3}i\right)^6
  \)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \[
    u = 1+\sqrt{3}i
    \qquad\qquad
    \pause a = \Re(u) = 1
    \qquad\qquad
    \pause b = \Im(u) = \sqrt{3}
  \]
  \[
    \pause \rho
    \pause = \abs{u}
    \pause = \sqrt{a^2 + b^2}
    \pause = \sqrt{1 + 3}
    \pause = \sqrt{4}
    \pause = 2
  \]
  \[
    \pause \sin(\varphi)
    \pause = \frac{b}{\rho}
    \pause = \frac{\sqrt{3}}{2}
  \]
  \[
    \pause \cos(\varphi)
    \pause = \frac{a}{\rho}
    \pause = \frac{1}{2}
  \]
  \[
    \pause \varphi
    \pause = \arg(u)
    \pause = \ang{60}
    \pause = \frac{\pi}{3} \si{\radian}
  \]
  \[
    \pause u
    \pause = \rho\left(\cos(\varphi)+i\sin(\varphi)\right)
    \pause = 2\left(\cos\left(\frac{\pi}{3}\right)+i\sin\left(\frac{\pi}{3}\right)\right)
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \onslide<+->{
  \[
    u
    = 1+\sqrt{3}i
    = 2\left(\cos\left(\frac{\pi}{3}\right)+i\sin\left(\frac{\pi}{3}\right)\right)
  \]}
  \vspace{-\baselineskip}
  \begin{align*}
    \onslide<+->{z = u^6}
    \onslide<+->{& = \rho^6 \big[\cos(6\varphi) + i\sin(6\varphi)\big] \\}
    \onslide<+->{& = 2^6 \left[\cos\left(6\frac{\pi}{3}\right) + i\sin\left(6\frac{\pi}{3}\right)\right] \\}
    \onslide<+->{& = 64  \left[\cos\left(2\pi\right) + i\sin\left(2\pi\right)\right] \\}
    \onslide<+->{& = 64  \left[1 + i0\right] \\}
    \onslide<+->{& = 64}
\end{align*}
\end{resolution}
