%------------------------------------------------------------------------------%
\begin{question}
  Encontre as raízes cúbicas complexas do número \( z = -27 \)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  Escrevemos \( z = -27 \) na forma polar
  \[
    z
    = 27 [1 + i0]
    = 27 \left[ \cos\left(\pi\right) + i \sin\left(\pi\right) \right]
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Raízes cúbicas}
\begin{align*}
  u_k
  & = \sqrt[3]{27} \left[
  \cos\left(\frac{\pi+2k\pi}{3}\right)
  + i \sin\left(\frac{\pi+2k\pi}{3}\right)
  \right]
  & k = 0, 1, 2    \\
  & = 3 \left[
  \cos\left(\frac{\pi+2k\pi}{3}\right)
  + i \sin\left(\frac{\pi+2k\pi}{3}\right)
  \right]
\end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{\(k=0\)}
\begin{align*}
  u_0
  & = 3 \left[
  \cos\left( \frac{\pi+2\times 0\pi}{3} \right)
  + i \sin\left( \frac{\pi+2\times 0\pi}{3} \right)
  \right] \\
  & = 3 \left[
  \cos\left( \frac{\pi}{3} \right)
  + i \sin\left( \frac{\pi}{3} \right)
  \right] \\
  & = 3 \left[
  \cos\left( \ang{60} \right)
  + i \sin\left( \ang{60} \right)
  \right] \\
  & = 3 \left[
  \frac{1}{2}
  + i \frac{\sqrt{3}}{2}
  \right] \\
  & =  \frac{3}{2} + \frac{3\sqrt{3}}{2} i
\end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{\(k=1\)}
\begin{align*}
  u_1
  & = 3 \left[
  \cos\left( \frac{\pi+2\times 1\pi}{3} \right)
  + i \sin\left( \frac{\pi+2\times 1\pi}{3} \right)
  \right] \\
  & = 3 \left[
  \cos\left( \pi \right)
  + i \sin\left( \pi \right)
  \right] \\
  & = 3 \left[ -1 + i0 \right] \\
  & = -3
\end{align*}

\end{resolution}

  \right] \\
  & = 3 \left[

  \right] \\

\end{align*}
\end{resolution}
