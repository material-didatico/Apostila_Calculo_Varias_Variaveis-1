\section{Exponencial Complexa}

%------------------------------------------------------------------------------%
\begin{proofframe}{Justificativa}
  Escolhendo $z=i\theta$
  \begin{align*}
    \action<+->{e^{i\theta}}
    \action<+->{& = 1 + (i\theta)
      + \frac{(i\theta)^2}{2!}
      + \frac{(i\theta)^3}{3!}
      + \frac{(i\theta)^4}{4!}
      + \frac{(i\theta)^5}{5!}
      + \frac{(i\theta)^6}{6!}
      + \cdots \\[2mm]}
    \action<+->{& = 1 +  i\theta
      + \frac{i^2\theta^2}{2!}
      + \frac{i^3\theta^3}{3!}
      + \frac{i^4\theta^4}{4!}
      + \frac{i^5\theta^5}{5!}
      + \frac{i^6\theta^6}{6!}
      + \cdots \\[2mm]}
    \action<+->{& = 1 +  i\theta
      - \frac{\theta^2}{2!}
      - i\frac{\theta^3}{3!}
      + \frac{\theta^4}{4!}
      + i\frac{\theta^5}{5!}
      - \frac{\theta^6}{6!}
      + \cdots \\[2mm]}
    \action<+->{& = \left( 1 - \frac{\theta^2}{2!}
      + \frac{\theta^4}{4!}
      - \frac{\theta^6}{6!}
      + \cdots
      \right)
      + i \left( \theta - \frac{\theta^3}{3!}
      + \frac{\theta^5}{5!}
      - \cdots
      \right) \\[2mm]}
    \action<+->{& = \cos\theta + i\sin\theta}
  \end{align*}
\end{proofframe}
