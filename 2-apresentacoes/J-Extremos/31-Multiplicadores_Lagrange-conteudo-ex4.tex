%------------------------------------------------------------------------------%
\begin{question}
  Encontre o ponto do plano \(x+2y+3z=13\) mais próximo do ponto \((1, 1, 1)\)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Formulação do Problema}
  Queremos minimizar a distância
  \[
    d(x, y, z) = \sqrt{(x-1)^2 + (y-1)^2 + (z-1)^2}
  \]
  \pause
  Como a função raiz quadrada é crescente podemos minimizar a função
  \[
    f(x, y, z) = (x-1)^2 + (y-1)^2 + (z-1)^2
  \]
  \pause
  Sujeito a condição
  \[
    g(x, y, z) = x + 2y + 3z = 13
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Gradientes}
\begin{columns}
\column{0.5\linewidth}
  \onslide<+->{\(f(x, y, z) = (x-1)^2 + (y-1)^2 + (z-1)^2\)}
  \begin{align*}
    \onslide<+->{f_x(x, y, z) & = 2(x-1) \\}
    \onslide<+->{f_y(x, y, z) & = 2(y-1) \\}
    \onslide<+->{f_z(x, y, z) & = 2(z-1)}
  \end{align*}
  \[
    \onslide<+->{\nabla f = \vetortri{2(x-1)}{2(y-1)}{2(z-1)}}
  \]
\column{0.5\linewidth}
  \onslide<+->{\(g(x, y, z) = x + 2y + 3z\)}
  \begin{align*}
    \onslide<+->{g_x(x, y, z) & = 1 \\}
    \onslide<+->{g_y(x, y, z) & = 2 \\}
    \onslide<+->{g_z(x, y, z) & = 3}
  \end{align*}
  \[
    \onslide<+->{\nabla g = \vetortri{1}{2}{3}}
  \]
\end{columns}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Sistema}
  Aplicando os Multiplicadores de Lagrange, precisamos resolver o sistema
  \[
  \begin{cases}
    2(x-1) =  \lambda \\
    2(y-1) = 2\lambda \\
    2(z-1) = 3\lambda \\
    x + 2y + 3z = 13
  \end{cases}
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  Das três primeiras equações obtemos
  \begin{columns}
  \column{0.3\linewidth}
    \[
      \pause
      2(x-1) = \lambda
    \]
    \[
      \pause
      x = \frac{\lambda}{2} + 1
    \]
  \column{0.3\linewidth}
    \[
      \pause
      2(y-1) = 2\lambda
    \]
    \[
      \pause
      y = \lambda + 1
    \]
  \column{0.3\linewidth}
    \[
      \pause
      2(z-1) = 3\lambda
    \]
    \[
      \pause
      z = \frac{3\lambda}{2} + 1
    \]
  \end{columns}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  Substituindo na última equação
  \begin{align*}
    \onslide<+->{x + 2y + 3z & = 13}
    \\
    \onslide<+->{\left(\frac{\lambda}{2} + 1\right)
             + 2 \left(\lambda + 1\right)
             + 3 \left(\frac{3\lambda}{2} + 1\right) & = 13}
    \\
    \onslide<+->{\frac{\lambda}{2} + 1
              + 2\lambda + 2
              + \frac{9\lambda}{2} + 3 & = 13}
    \\
    \onslide<+->{\lambda + 2
              + 4\lambda + 4
              + 9\lambda + 6 & = 26}
    \\
    \onslide<+->{14\lambda + 12 & = 26}
    \\
    \onslide<+->{14\lambda & = 14}
    \\
    \onslide<+->{\lambda & = 1}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  Portanto
  \[
    \pause x = \frac{\lambda}{2} + 1
    \pause = \frac{1}{2} + 1
    \pause = \frac{3}{2}
  \]

  \[
    \pause y = \lambda + 1
    \pause = 1 + 1
    \pause = 2
  \]

  \[
    \pause z = \frac{3\lambda}{2} + 1
    \pause = \frac{3\times 1}{2} + 1
    \pause = \frac{5}{2}
  \]

  \pause
  O ponto mais próximo é o ponto \(\left(\frac{3}{2}, 2, \frac{5}{2}\right)\)
\end{resolution}
