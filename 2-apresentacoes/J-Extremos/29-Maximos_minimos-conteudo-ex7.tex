\begin{question}
  Considerando a funﾃｧﾃ｣o
  \(
    f(x, y) = x^3 + 3xy + y^3
  \)

  a) Calcule o gradiente de $f$

  b) Calcule a hessiana de $f$

  c) Encontre todos os pontos crﾃｭticos de $f$

  d) Classifique cada ponto crﾃｭtico de $f$
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{a}
  Precisamos das derivadas parciais de $f$
  \[
    \D{f}{x}
    = \D{}{x}\left[x^3 + 3xy + y^3\right]
    = 3x^2 + 3y
  \]
  \[
    \D{f}{y}
    = \D{}{y}\left[x^3 + 3xy + y^3\right]
    = 3x + 3y^2
  \]
  entﾃ｣o
  \[
    \nabla f = \vetor{3x^2 + 3y}{3x + 3y^2}
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{b}
  Precisamos das derivadas parciais de segunda ordem de $f$
  \[
    \DS{f}{x}
    = \D{}{x}\left[3x^2 + 3y\right]
    = 6x
  \]
  \[
    \D{f}{y}
    = \D{}{y}\left[3x + 3y^2\right]
    = 6y
  \]
  \[
    \DM{f}{x}{y}
    = \D{}{x}\D{f}{y}
    = \D{}{x}\left[3x + 3y^2\right]
    = 3
  \]
  entﾃ｣o
  \[
    H = \left(\begin{array}{cc}
      6x & 3 \\
      3 & 6y
    \end{array}\right)
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{c}
  A funﾃｧﾃ｣o ﾃｩ um polinﾃｴmio, entﾃ｣o possui derivadas em todos os pontos do plano.
  Assim os pontos crﾃｭticos sﾃ｣o apenas os pontos onde as derivadas parciais sﾃ｣o zero,
  \(\nabla f = 0\)
  \[
    3x^2 + 3y = 0
    \qquad\text{e}\qquad
    3x + 3y^2 = 0
  \]
  ou, simplificando,
  \[
    x^2 + y = 0
    \qquad\text{e}\qquad
    x + y^2 = 0
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{c}
  Isolando $y$ na primeira equaﾃｧﾃ｣o, \(y = -x^2\), e substituindo na segunda, temos
  \begin{align*}
    x + y^2                 & = 0 \\
    x + \left(-x^2\right)^2 & = 0 \\
    x + x^4                 & = 0 \\
    x(1+ x^3)               & = 0
  \end{align*}
  As soluﾃｧﾃｵes dessa equaﾃｧﾃ｣o sﾃ｣o $x=0$ ou $x-1$.
  Se $x=0$ temos $y=0$ e
  se $x=-1$ temos $y=-1$.
  Portanto, os pontos crﾃｭticos sﾃ｣o
  \[
    (x_1, y_1) = (0, 0)
    \qquad\text{e}\qquad
    (x_2, y_2) = (-1, -1)
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{d}
  Para classificar os pontos crﾃｭticos precisamos avaliar o discriminante nos
  pontos crﾃｭticos.
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{d}
  Considerando o ponto \((x_1, y_1) = (0, 0)\)
  \[
    f_{xx}(0, 0) = 0
  \]
  \[
    f_{yy}(0, 0) = 0
  \]
  \[
    f_{xy}(0, 0) = 3
  \]
  \[
    D_1
    = f_{xx}(0, 0)f_{yy}(0, 0) - f^2_{xy}(0, 0)
    = 0 \times 0 - 3^2
    = -9 < 0
  \]
  Portanto, o ponto $(0, 0)$ ﾃｩ um ponto de sela.
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{d}
  Considerando o ponto \((x_2, y_2) = (-1, -1)\)
  \[
    f_{xx}(-1, -1) = -6
  \]
  \[
    f_{yy}(-1, -1) = -6
  \]
  \[
    f_{xy}(-1, -1) = 3
  \]
  \[
    D_2
    = f_{xx}(-1, -1)f_{yy}(-1, -1) - f^2_{xy}(-1, -1)
    = (-6)(-6) - 3^2
    = 36 - 9
    = 25 > 0
  \]
  Portanto, o ponto $(-1, -1)$ ﾃｩ um mﾃ｡ximo ou mﾃｭnimo local.
  Como \(f_{xx}(-1, -1) = -6 < 0\)
  o ponto ﾃｩ um ponto de mﾃ｡ximo local.
\end{resolution}
