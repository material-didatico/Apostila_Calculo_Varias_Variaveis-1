%------------------------------------------------------------------------------%
\begin{question}
  Encontre o ponto da esfera \( x^2 + y^2 + z^2 = 4 \)
  mais distante do ponto \((1, -1, 1)\)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Formulação do Problema}
  Queremos maximizar a distância
  \[
    d(x, y, z) = \sqrt{(x-1)^2 + (y+1)^2 + (z-1)^2}
  \]
  \pause
  Como a função raiz quadrada é crescente, podemos maximizar a função
  \[
    f(x, y, z) = (x-1)^2 + (y+1)^2 + (z-1)^2
  \]
  \pause
  sujeito a restrição
  \[
    g(x, y, z) = x^2 + y^2 + z^2 = 4
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Gradientes}
\begin{columns}
\column{0.55\linewidth}
  \onslide<+->{\(f(x, y, z) = (x-1)^2 + (y+1)^2 + (z-1)^2\)}
  \begin{align*}
    \onslide<+->{f_x(x, y, z) & = 2(x-1) \\}
    \onslide<+->{f_y(x, y, z) & = 2(y+1) \\}
    \onslide<+->{f_z(x, y, z) & = 2(z-1)}
  \end{align*}
  \[
    \onslide<+->{\nabla f = \vetortri{2(x-1)}{2(y+1)}{2(z-1)}}
  \]
\column{0.45\linewidth}
  \onslide<+->{\(g(x, y, z) = x^2 + y^2 + z^2\)}
  \begin{align*}
    \onslide<+->{g_x(x, y, z) & = 2x \\}
    \onslide<+->{g_y(x, y, z) & = 2y \\}
    \onslide<+->{g_z(x, y, z) & = 2z}
  \end{align*}
  \[
    \onslide<+->{\nabla g = \vetortri{2x}{2y}{2z}}
  \]
\end{columns}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Sistema}
  Aplicando os Multiplicadores de Lagrange, precisamos resolver o sistema
  \[
  \begin{cases}
    2(x-1) = 2\lambda x \\
    2(y+1) = 2\lambda y \\
    2(z-1) = 2\lambda z \\
    x^2 + y^2 + z^2 = 4
  \end{cases}
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \onslide<+->{Das três primeiras equações obtemos}
  \begin{columns}
  \column{0.3\linewidth}
    \begin{align*}
      \onslide<+->{2(x-1)       & = 2\lambda x        \\}
      \onslide<+->{x-1          & = \lambda x         \\}
      \onslide<+->{x(1-\lambda) & = 1                 \\}
      \onslide<+->{x            & = \frac{1}{1-\lambda} }
    \end{align*}
  \column{0.3\linewidth}
  \begin{align*}
    \onslide<+->{2(y+1)       & = 2\lambda y        \\}
    \onslide<+->{y+1          & = \lambda y         \\}
    \onslide<+->{y(1-\lambda) & = -1                \\}
    \onslide<+->{y            & =\frac{-1}{1-\lambda} }
  \end{align*}
  \column{0.3\linewidth}
  \begin{align*}
    \onslide<+->{2(z-1)       & = 2\lambda z       \\}
    \onslide<+->{z-1          & = \lambda z        \\}
    \onslide<+->{z(1-\lambda) & = 1                \\}
    \onslide<+->{z            & = \frac{1}{1-\lambda}}
  \end{align*}
  \end{columns}

  \vspace{\baselineskip}
  \onslide<+->{Usamos que \(1-\lambda\neq 0\) pois \(x(1-\lambda) = 1\)}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Substituindo na Última Equação}
  \begin{align*}
    \onslide<+->{x^2 + y^2 + z^2 & = 4}
    \\
    \onslide<+->{\left(\frac{1}{1-\lambda}\right)^2 +
                 \left(\frac{-1}{1-\lambda}\right)^2 +
                 \left(\frac{1}{1-\lambda}\right)^2 & = 4}
    \\
    \onslide<+->{\frac{1 + 1 + 1}{\left(1-\lambda\right)^2} & = 4}
    \\
    \onslide<+->{\frac{1}{\left(1-\lambda\right)^2} & = \frac{4}{3}}
    \\
    \onslide<+->{\frac{1}{1-\lambda} & = \pm\sqrt{\frac{4}{3}}}
    \onslide<+->{\;=\; \pm\frac{2}{\sqrt{3}}}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
\begin{columns}
\column{0.4\linewidth}
  Para \(\frac{1}{1-\lambda} = \frac{2}{\sqrt{3}}\)
  \[
    \pause x = \frac{1}{1-\lambda}
    \pause   = \frac{2}{\sqrt{3}}
  \]

  \[
    \pause y = \frac{-1}{1-\lambda}
    \pause   = \frac{-2}{\sqrt{3}}
  \]

  \[
    \pause z = \frac{1}{1-\lambda}
    \pause   = \frac{2}{\sqrt{3}}
  \]
\column{0.4\linewidth}
  \pause
  Para \(\frac{1}{1-\lambda} = -\frac{2}{\sqrt{3}}\)
  \[
    \pause x = \frac{1}{1-\lambda}
    \pause   = \frac{-2}{\sqrt{3}}
  \]

  \[
    \pause y = \frac{-1}{1-\lambda}
    \pause   = \frac{2}{\sqrt{3}}
  \]

  \[
    \pause z = \frac{1}{1-\lambda}
    \pause   = \frac{-2}{\sqrt{3}}
  \]
\end{columns}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Candidatos a Máximo}
  Temos dois candidatos
  \(\left( \frac{2}{\sqrt{3}}, \frac{-2}{\sqrt{3}}, \frac{2}{\sqrt{3}}\right)\) e
  \(\left( \frac{-2}{\sqrt{3}}, \frac{2}{\sqrt{3}}, \frac{-2}{\sqrt{3}}\right)\)
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Avaliando a Função}
  \begin{align*}
    \onslide<+->{f\left( \frac{2}{\sqrt{3}}, \frac{-2}{\sqrt{3}}, \frac{2}{\sqrt{3}}\right)}
    \onslide<+->{& = \left(\frac{ 2}{\sqrt{3}}-1\right)^2
    + \left(\frac{-2}{\sqrt{3}}+1\right)^2
    + \left(\frac{ 2}{\sqrt{3}}-1\right)^2 \\[2mm]}
    \onslide<+->{& = \left(\frac{2}{\sqrt{3}}-1\right)^2
    + \left(\frac{2}{\sqrt{3}}-1\right)^2
    + \left(\frac{2}{\sqrt{3}}-1\right)^2 \\[2mm]}
    \onslide<+->{& = 3\left(\frac{2}{\sqrt{3}}-1\right)^2}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Avaliando a Função}
  \begin{align*}
    \onslide<+->{f\left( \frac{-2}{\sqrt{3}}, \frac{2}{\sqrt{3}}, \frac{-2}{\sqrt{3}}\right)}
    \onslide<+->{& = \left(\frac{-2}{\sqrt{3}}-1\right)^2
    + \left(\frac{ 2}{\sqrt{3}}+1\right)^2
    + \left(\frac{-2}{\sqrt{3}}-1\right)^2 \\[2mm]}
    \onslide<+->{& = \left(\frac{2}{\sqrt{3}}+1\right)^2
    + \left(\frac{2}{\sqrt{3}}+1\right)^2
    + \left(\frac{2}{\sqrt{3}}+1\right)^2 \\[2mm]}
    \onslide<+->{& = 3\left(\frac{2}{\sqrt{3}}+1\right)^2}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  Como
  \[
    f\left( \frac{2}{\sqrt{3}}, \frac{-2}{\sqrt{3}}, \frac{2}{\sqrt{3}}\right)
    = 3\left(\frac{2}{\sqrt{3}}-1\right)^2
  \]
  é menor do que
  \[
    f\left( \frac{-2}{\sqrt{3}}, \frac{2}{\sqrt{3}}, \frac{-2}{\sqrt{3}}\right)
    = 3\left(\frac{2}{\sqrt{3}}+1\right)^2
  \]

  \pause
  O ponto mais distante é
  \(\left( \frac{-2}{\sqrt{3}}, \frac{2}{\sqrt{3}}, \frac{-2}{\sqrt{3}}\right)\)
  \clearpage
\end{resolution}
