%------------------------------------------------------------------------------%
\begin{question}
  Encontre todas as coordenadas polares associadas ao ponto
  \(
    \left(r, \theta\right) =
    \left(2, \frac{\pi}{6}\right)
  \)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  O ponto
  \(
    \left(r, \theta\right) =
    \left(2, \frac{\pi}{6}\right)
  \)
  também pode ser escrito como
  \[
    \left(r, \theta\right) =
    \left(\hphantom{-} 2, \hphantom{-5}\,\frac{\pi}{6}+2n\pi\right)
    \qquad n=0, \pm 1, \pm 2, \dots
  \]
  ou
  \[
    \left(r, \theta\right) =
    \left(-2, -\frac{5\pi}{6}+2n\pi\right)
    \qquad n=0, \pm 1, \pm 2, \dots
  \]
\end{resolution}
