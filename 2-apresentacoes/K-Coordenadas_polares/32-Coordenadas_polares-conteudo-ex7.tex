\begin{question}
  Escreva a equação
  \[
    r = 2\cos(\theta) - \sin(\theta)
  \]
  em coordenadas cartesianas e simplifique
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \onslide<+->{Sabemos que
  \[
    \cos(\theta) = \frac{x}{r}
    \qquad\qquad
    \sin(\theta) = \frac{y}{r}
    \qquad\qquad
    r^2 = x^2 + y^2
  \]}
\begin{columns}
\column{0.3\linewidth}
  \begin{align*}
    \onslide<+->{r & = 2\cos(\theta) - \sin(\theta) \\}
    \onslide<+->{r & = 2\frac{x}{r} - \frac{y}{r}   \\}
    \onslide<+->{r & = \frac{2x-y}{r}}
  \end{align*}
\column{0.5\linewidth}
  \begin{align*}
    \onslide<+->{r^2               & = 2x-y \\}
    \onslide<+->{x^2 + y^2         & = 2x-y \\}
    \onslide<+->{x^2 + y^2 - 2x+ y & = 0}
  \end{align*}
\end{columns}
\end{resolution}
