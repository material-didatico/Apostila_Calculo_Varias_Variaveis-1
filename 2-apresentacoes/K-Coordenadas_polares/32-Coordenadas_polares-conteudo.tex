\begin{frame}{Coordenadas Polares}
  Definição
  \begin{itemize}[<+->]
    \item Origem ou polo
    \item Raio inicial
    \item \emph{$r$ distância orientada} até a origem, \(r\in\R\)
    \item \emph{$\theta$ ângulo orientado} a partir do raio inicial
          (em radianos) \(\theta\in\R\)
    \item Coordenadas polares $(r, \theta)$
  \end{itemize}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Equações em Coordenadas Polares}
  \onslide<+->{\(r=a\)}
  \tabto{20mm}
  \onslide<+->{circulo com centro na origem e raio \(\abs{a}\)}
  \\[9mm]

  \onslide<+->{\(\theta=\theta_0\)}
  \tabto{20mm}
  \onslide<+->{reta passando pela origem com ângulo $\theta_0$ com o raio inicial}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Relação Entre Coordenadas Polares e Cartesianas}
\begin{columns}
\column{0.2\linewidth}
\begin{align*}
  \onslide<2->{x          & = r\cos(\theta) \\}
  \onslide<3->{y          & = r\sin(\theta) \\[9mm]}
  \onslide<4->{r^2        & = x^2 + y^2     \\}
  \onslide<5->{\tan\theta & = \frac{x}{y}}
\end{align*}
\column{0.7\linewidth}
  \centering
  \onslide<1->{\includegraphics[width=\linewidth]{relacao_polares_catesianas}}
\end{columns}
\end{frame}
