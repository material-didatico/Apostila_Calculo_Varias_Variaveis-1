%------------------------------------------------------------------------------%
\begin{question}
  Escreva a equação
  \[
    r = \frac{4}{2\cos(\theta)-\sin(\theta)}
  \]
  em coordenadas cartesianas
  e identifique seu gráfico
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  Sabemos que
  \(
    x = r\cos(\theta) \quad
    y = r\sin(\theta)
  \)

  \[
    \pause
    r = \frac{4}{2\cos(\theta)-\sin(\theta)}
  \]
  \[
    \pause
    2r\cos(\theta) - r\sin(\theta) = 4
  \]
  \[
    \pause
    2x - y = 4
  \]
  \[
    \pause
    y = 2x - 4
  \]

  \pause
  O gráfico é uma reta
\end{resolution}
