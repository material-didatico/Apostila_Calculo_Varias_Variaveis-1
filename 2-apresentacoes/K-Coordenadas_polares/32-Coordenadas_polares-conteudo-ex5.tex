\begin{question}
  Escreva a equação
  \[
    r = \frac{4}{2\cos(\theta)-\sin(\theta)}
  \]
  em coordenadas cartesianas
  e identifique seu gráfico
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \onslide<+->{Sabemos que
  \(
    x = r\cos(\theta) \quad
    y = r\sin(\theta)
  \)}
  \begin{align*}
    \onslide<+->{r = \frac{4}{2\cos(\theta)-\sin(\theta)}     &   \\}
    \onslide<+->{2r\left(\cos(\theta) - \sin(\theta)\right) = & \;4 \\}
    \onslide<+->{2r\cos(\theta) - r\sin(\theta)             = & \;4 \\}
    \onslide<+->{2x - y                                     = & \;4 \\}
    \onslide<+->{y                                          = & \;2x - 4}
  \end{align*}
  \onslide<+->{O gráfico é uma reta}
\end{resolution}
