\begin{question}
  Escreva a equação
  \[
    r = \frac{5}{\sin(\theta)-2\cos(\theta)}
  \]
  em coordenadas cartesianas e simplifique
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \onslide<+->{Sabemos que
  \[
    \cos(\theta) = \frac{x}{r}
    \qquad\qquad
    \sin(\theta) = \frac{y}{r}
    \qquad\qquad
    r^2 = x^2 + y^2
  \]}
\begin{columns}
\column{0.3\linewidth}
  \begin{align*}
    \onslide<+->{r = \frac{5}{\sin(\theta)-2\cos(\theta)}   &   \\}
    \onslide<+->{r\left(\sin(\theta)-2\cos(\theta)\right) = & \;5 \\}
    \onslide<+->{r\left(\frac{y}{r}-2\frac{x}{r}\right)   = & \;5}
  \end{align*}
\column{0.5\linewidth}
  \begin{align*}
    \onslide<+->{r\frac{y-2x}{r} & = 5      \\}
    \onslide<+->{y-2x            & = 5      \\}
    \onslide<+->{y               & = 2x + 5}
  \end{align*}
\end{columns}
\end{resolution}
