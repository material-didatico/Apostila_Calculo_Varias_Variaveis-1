\begin{question}
  Escreva a equação
  \[
    r = \frac{5}{\sin(\theta)-2\cos(\theta)}
  \]
  em coordenadas cartesianas e simplifique
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \onslide<+->{Sabemos que
  \[
    x = r\cos(\theta)
    \qquad\qquad
    y = r\sin(\theta)
    \qquad\qquad
    r^2 = x^2 + y^2
  \]}
  \vspace{-1.3\baselineskip}
  \begin{align*}
    \onslide<+->{r = \frac{5}{\sin(\theta)-2\cos(\theta)}   &     \\}
    \onslide<+->{r\left(\sin(\theta)-2\cos(\theta)\right) = & \;5 \\}
    \onslide<+->{r\sin(\theta)-2r\cos(\theta)             = & \;5 \\}
    \onslide<+->{y-2x = & \; 5   \\}
    \onslide<+->{y    = & \; 2x + 5}
  \end{align*}
\end{resolution}
