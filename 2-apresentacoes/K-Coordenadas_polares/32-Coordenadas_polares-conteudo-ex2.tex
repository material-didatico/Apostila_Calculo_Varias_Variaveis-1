\begin{question}
  Converta as seguintes coordenadas cartesianas em coordenadas polares
  $(r, \theta)$

  a) \quad \((x, y) = \left(3, 4\right)\)

  b) \quad \((x, y) = \left(-2, 2\sqrt{3}\right)\)

  c) \quad \((x, y) = \left(0, -5\right)\)

  Apresente as coordenadas com os valores ``mais simples''
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução Item a}
  \[
    r^2    \;=\; x^2 + y^2
    \pause \;=\; 3^2 + 4^2
    \pause \;=\; 9 + 16
    \pause \;=\; 25
  \]
  \[
    \pause r = \pm 5
  \]

  \[
    \pause \tan\theta \;=\; \frac{y}{x}
    \pause            \;=\; \frac{4}{3}
  \]
  \[
    \pause \theta \;=\; \arctan\left(\frac{4}{3}\right) \pm n\pi
    \qquad n\in \Z
  \]
  \[
    \pause
    \left(r, \theta\right) =
    \left(5, \arctan\left(\frac{4}{3}\right)\right)
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução Item b}
  \[
    r^2    \;=\; x^2 + y^2
    \pause \;=\; (-2)^2 + \left(2\sqrt{3}\right)^2
    \pause \;=\; 4 + 4\times 3
    \pause \;=\; 16
  \]
  \[
    \pause r = \pm 4
  \]

  \[
    \pause \tan\theta \;=\; \frac{y}{x}
    \pause            \;=\; \frac{2\sqrt{3}}{2}
    \pause            \;=\; \sqrt{3}
  \]
  \[
    \pause \theta \;=\; \arctan\left(\sqrt{3}\right) \pm n\pi
    \pause        \;=\; \frac{\pi}{3} \pm n\pi
    \qquad n\in \Z
  \]

  \pause
  \[
    \left(r, \theta\right) =
    \left(4, \dfrac{2\pi}{3}\right)
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução Item c}
  \[
    r^2    \;=\; x^2 + y^2
    \pause \;=\; 0^2 + 5^2
    \pause \;=\; 25
  \]
  \[
    \pause r = \pm 5
  \]

  \[
    \pause \tan\theta \;=\; \frac{y}{x}
    \pause            \;=\; \frac{-5}{0}
    \pause            \quad \nexists
  \]
  \[
    \pause \theta \;=\; \frac{\pi}{2} \pm n\pi
    \qquad n\in \Z
  \]

  \[
    \pause
    \left(r, \theta\right) =
    \left(5, \dfrac{3\pi}{2}\right)
  \]
\end{resolution}
