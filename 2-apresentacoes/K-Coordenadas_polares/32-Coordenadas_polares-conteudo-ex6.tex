\begin{question}
  Mostre que a equação polar
  \[
    r = \frac{4}{2 - \cos\theta}
  \]
  representa uma cônica
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
\begin{columns}[t]
\column{0.3\linewidth}
  \begin{align*}
    \onslide<+->{r                             & = \frac{4}{2 - \cos\theta} \\}
    \onslide<+->{\left(2 - \cos\theta\right)r  & = 4                        \\}
    \onslide<+->{\left(2 - \frac{x}{r}\right)r & = 4                        \\}
    \onslide<+->{2r - x                        & = 4                        \\}
    \onslide<+->{2r                            & = 4 + x                    \\}
  \end{align*}
\column{0.6\linewidth}
  \begin{align*}
    \onslide<+->{4r^2                          & = (4 + x)^2                \\}
    \onslide<+->{4\left(x^2+y^2\right)         & = 16 +8x + x^2             \\}
    \onslide<+->{4x^2 +4y^2 -8x -x^2           & = 16                       \\[7mm]}
    \onslide<+->{3x^2 +4y^2 -8x                & = 16}
  \end{align*}
\end{columns}
\end{resolution}
