\section{Cônicas}

\begin{frame}{Superfícies Quádricas}

  \emph{Cônica}:
  Gráfico de uma equação de segundo grau no plano
  \[
    Ax^2 + By^2 + Cxy + Dx + Ey + F = 0
  \]

  \emph{Quádrica}:
  gráfico de uma equação de segundo grau em $\R^3$
  \[
    Ax^2 + By^2 + Cz^2 +
    Dxy  + Exz  + Fyz +
    Gx   + Hy   + Iz + G = 0
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Cônicas}
  Seções cônicas
  \begin{itemize}
    \item Elipse (círculo)
    \item Parábola
    \item Hipérbole
  \end{itemize}
  Seção 11.6 do livro do Thomas
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Parábola}
  \begin{columns}
    \column{0.4\linewidth}
    Pontos equidistantes de um ponto fixo (foco)
    e uma reta dada (diretriz)
    \[
      y = \frac{x^2}{4p}
    \]
    \column{0.6\linewidth}
    \centering
    \includegraphics[width=\linewidth]{parabola}
  \end{columns}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Elipse}
  \begin{columns}
    \column{0.4\linewidth}
    Pontos cuja soma das distâncias até
    os focos é constante
    \[
      \frac{x^2}{a^2} + \frac{y^2}{b^2} = 1
    \]
    \column{0.6\linewidth}
    \centering
    \includegraphics[width=0.8\linewidth]{elipse}
  \end{columns}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Hipérbole}
  \begin{columns}
    \column{0.4\linewidth}
    Pontos cujas distâncias a partir de dois pontos fixos (focos)
    tem diferença constante
    \[
      \frac{x^2}{a^2} - \frac{y^2}{b^2} = 1
    \]
    A hipérbole tem duas assintotas
    \column{0.6\linewidth}
    \centering
    \includegraphics[width=0.8\linewidth]{hiperbole}
  \end{columns}
\end{frame}
