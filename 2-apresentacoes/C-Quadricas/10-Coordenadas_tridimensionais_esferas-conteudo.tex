\begin{frame}{Distância}
  Distancia entre dois pontos \(P_1(x_1, y_1, z_1)\) e  \(P_2(x_2, y_2, z_2)\)
  \[
    \dist(P_1, P_2)
    = \abs{P_1P_2}
    = \sqrt{\left(x_2-x_1\right)^2 + \left(y_2-y_1\right)^2 + \left(z_2-z_1\right)^2}
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo 1}
  Interprete geometricamente
  \begin{enumerate}
    \item \(z > 0\)
    \item \(x = -3\)
    \item \(z=0\), \(x\leq 0\), \(y\geq 0\)
    \item \(-1\leq y \leq 1\)
    \item \(y=-2\), \(z=2\)
  \end{enumerate}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo 1}
  \begin{enumerate}[<+->]
    \item \(z > 0\) \\
          Semiespaço dos pontos acima do plano $z$
    \item \(x = -3\) \\
          Plano perpendicular ao eixo $x$ que passa pelo ponto $x=-3$
    \item \(z=0\), \(x\leq 0\), \(y\geq 0\) \\
          Segundo quadrante do plano $xy$
    \item \(-1\leq y \leq 1\) \\
          Fatia entre os planos \(y=-1\) e \(y=1\)
    \item \(y=-2\), \(z=2\) \\
          Reta comum aos planos \(y=-2\) e \(z=2\)
  \end{enumerate}
\end{frame}

\section{Esferas}

%------------------------------------------------------------------------------%
\begin{frame}{Esferas}
  Equação padrão da esfera de centro \((x_0, y_0, z_0)\) e raio \(r\)
  \[
    \left(x-x_0\right)^2 + \left(y-y_0\right)^2 + \left(z-z_0\right)^2 = r^2
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo 2}
  Encontre o centro da esfera
  \[
    x^2 + y^2 + z^2 + 6x - 4z + 1 = 0
  \]
  \pause
  Dica
  \[
    (a+b)^2 = a^2 + 2ab + b^2
  \]
  \[
    (a-b)^2 = a^2 - 2ab + b^2
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Exemplo 2}
  \onslide<+->{Completando os quadrados}
  \begin{align*}
    \onslide<+->{x^2 + y^2 + z^2 + 6x - 4z + 1 & = 0 \\}
    \onslide<+->{\left(x^2 + 6x\right) + y^2 + \left(z^2 - 4z\right) & = -1 \\}
    \onslide<+->{\left(x^2 + 2\cdot 3x\right)}
    \onslide<+->{+ \left(y-0\right)^2}
    \onslide<+->{+ \left(z^2 - 2\cdot 2z\right)} &
    \onslide<+->{= -1} \\
    \onslide<+->{\left(x^2 + 2\cdot 3x + 3^2\right) - 3^2}
    \onslide<+->{+ \left(y-0\right)^2}
    \onslide<+->{+ \left(z^2 - 2\cdot 2z + 2^2\right) -2^2} &
    \onslide<+->{= -1 \\}
    \onslide<+->{\left(x^2 + 2\cdot 3x + 3^2\right)
    + \left(y - 0\right)^2
    + \left(z^2 - 2\cdot 2z + 2^2\right) & = -1 + 3^2 + 2^2 \\}
    \onslide<+->{\left(x + 3\right)^2 +
    \left(y - 0\right)^2 +
    \left(z - 2\right)^2 & = 12}
  \end{align*}
  \onslide<+->{O centro da esfera é o ponto \((-3, 0, 2)\)
  o raio é \(\sqrt{12}\)}
\end{frame}
