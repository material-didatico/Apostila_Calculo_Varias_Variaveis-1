\exercise{30\%}
Calcule o limite ou mostre que ele não existe
\(
  \lim_{(x, y)\to(0, 0)} \frac{5y^4\cos^2(x)}{x^4+y^4}
\)
% \clearpagequestiononly

\begin{answer}
  Escolhendo o caminho $y=0$ temos
  \begin{align*}
    L_1
    & = \left(\lim_{(x, y)\to(0, 0)} \frac{5y^4\cos^2(x)}{x^4+y^4} \evalat{y=0}{} \right) \\[2mm]
    & = \lim_{x\to 0} \frac{5\times0\cos^2(x)}{x^4+0} \\[2mm]
    & = \lim_{x\to 0} \frac{0}{x^4} \\[2mm]
    & = \lim_{x\to 0} 0 \\[2mm]
    & = 0
  \end{align*}
  Escolhendo o caminho $x=0$ temos
  \begin{align*}
    L_2
    & = \left(\lim_{(x, y)\to(0, 0)} \frac{5y^4\cos^2(x)}{x^4+y^4} \evalat{x=0}{} \right) \\[2mm]
    & = \lim_{y\to 0} \frac{5y^4\cos^2(0)}{0+y^4} \\[2mm]
    & = \lim_{y\to 0} \frac{5y^4}{y^4} \\[2mm]
    & = \lim_{y\to 0} 5 \\[2mm]
    & = 5
  \end{align*}
  Limites diferentes por caminhos diferentes, portanto o limite não existe.
\end{answer}

\exercise{40\%}
Calcule \(\DM{f}{x}{y}\left(2, \sqrt{\pi}\right)\)\!,
respeitando a ordem descrita na notação, de
\(
  f(x, y) = \frac{\sin\left(xy^2\right)}{2xe^x}
\)
% \clearpagequestiononly

\begin{answer}
  \begin{align*}
    \D{f}{y}
    & = \D{}{y} \left[\frac{\sin\left(xy^2\right)}{2xe^x}\right]       \\[2mm]
    & = \frac{1}{2xe^x} \, \D{}{y} \left[\sin\left(xy^2\right)\right]  \\[2mm]
    & = \frac{1}{2xe^x} \cos\left(xy^2\right) \D{}{y}\left(xy^2\right) \\[2mm]
    & = \frac{1}{2xe^x} \cos\left(xy^2\right) x \D{y^2}{y}             \\[2mm]
    & = \frac{1}{2e^x} \cos\left(xy^2\right) 2y                        \\[2mm]
    & = \frac{y\cos\left(xy^2\right)}{e^x}
  \end{align*}
  \begin{align*}
    \DM{f}{x}{y}
    & = \D{}{x}\left(\D{f}{y}\right)                                  \\[2mm]
    & = \D{}{x}\left(\frac{y\cos\left(xy^2\right)}{e^x}\right)        \\[2mm]
    & = y \D{}{x}\left(\frac{\cos\left(xy^2\right)}{e^x}\right)       \\[2mm]
    & = \frac{y}{e^{2x}}\left[
           \D{}{x}\left[\cos\left(xy^2\right)\right]e^x  -
           \cos\left(xy^2\right)\D{e^x}{x}
        \right]                                                       \\[2mm]
    & = \frac{y}{e^{2x}}\left[
          -\sin\left(xy^2\right)\D{}{x}\left(xy^2\right)e^x  -
          \cos\left(xy^2\right)e^x
        \right]                                                       \\[2mm]
    & = \frac{-y}{e^{x}}\left[
          \sin\left(xy^2\right)y^2 + \cos\left(xy^2\right)
        \right]                                                       \\[2mm]
    & = \frac{-y}{e^{x}}\left[
        y^2\sin\left(xy^2\right) + \cos\left(xy^2\right)
        \right]
  \end{align*}
  \begin{align*}
    \DM{f}{x}{y}\left(2, \sqrt{\pi}\right)
    & = \frac{-y}{e^{x}}\left[
          y^2\sin\left(xy^2\right) + \cos\left(xy^2\right)
        \right] \evalat{\left(2, \sqrt{\pi}\right)}{} \\[2mm]
    & = \frac{-\sqrt{\pi}}{e^2}\left[
          \left(\sqrt{\pi}\right)^2\sin\left(2\left(\sqrt{\pi}\right)^2\right)
          + \cos\left(2\left(\sqrt{\pi}\right)^2\right)
        \right]  \\[2mm]
    & = \frac{-\sqrt{\pi}}{e^2}\left[
          \pi\sin\left(2\pi\right)
          + \cos\left(2\pi\right)
        \right]   \\[2mm]
    & = \frac{-\sqrt{\pi}}{e^2}\left[ 0 + 1 \right]    \\[2mm]
    & = \frac{-\sqrt{\pi}}{e^2}
  \end{align*}
\end{answer}

\exercise{30\%}
Assumindo que a equação
\(
  xy^2 + z^2 = 4xe^{yz}
  \vphantom{\int}
\)
define $z$ como função de $x$ e $y$,
utilize a derivação implícita para encontrar \(\D{z}{y}\)
% \clearpagequestiononly

\begin{answer}
  Utilizando a fórmula com \(F(x, y, z) = xy^2 + z^2 - 4xe^{yz}\)
  \begin{align*}
    F_y(x, y, z)
    & = \D{F}{y}(x, y, z) \\[2mm]
    & = \D{}{y} \left(xy^2 + z^2 - 4xe^{yz}\right) \\[2mm]
    & = \D{}{y} \left(xy^2\right)
    + \D{}{y} \left(z^2 \right)
    - \D{}{y} \left(4xe^{yz}\right) \\[2mm]
    & = x \D{y^2}{y} + 0 - 4x\D{}{y}\left(e^{yz}\right) \\[2mm]
    & = x2y - 4xe^{yz}\D{}{y}\left(yz\right) \\[2mm]
    & = 2xy - 4xe^{yz}z \\[2mm]
    & = 2xy - 4xze^{yz} \\[2mm]
    F_z(x, y, z)
    & = \D{F}{z}(x, y, z) \\[2mm]
    & = \D{}{z} \left(xy^2 + z^2 - 4xe^{yz}\right) \\[2mm]
    & = \D{}{z} \left(xy^2\right)
    + \D{}{z} \left(z^2 \right)
    - \D{}{z} \left(4xe^{yz}\right) \\[2mm]
    & = 0 + 2z - 4x\D{}{z}\left(e^{yz}\right) \\[2mm]
    & = 2z - 4xe^{yz}\D{}{z}\left(yz\right) \\[2mm]
    & = 2z - 4xye^{yz}
  \end{align*}
  Portanto,
  \[
  \D{z}{y}(x, y, z)
  = -\frac{F_y(x, y, z)}{F_z(x, y, z)}
  = -\frac{2xy - 4xze^{yz}}{2z - 4xye^{yz}}
  = \frac{2xze^{yz}- xy}{z - 2xye^{yz}}
  \]
\end{answer}
