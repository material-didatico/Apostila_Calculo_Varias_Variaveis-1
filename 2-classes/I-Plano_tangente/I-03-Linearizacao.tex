\section{Aproximação Linear}

%------------------------------------------------------------------------------%
\begin{frame}{Plano Tangente ao Gráfico de  $f(x, y)$}
  O plano tangente ao gráfico de
  de uma função diferenciável \(f(x, y)\) \\
  no ponto \((a, b)\) é
  \[
    \emph{f_x(a, b)} \big(x-a\big) +
    \emph{f_y(a, b)} \big(y-b\big) -
    \big(z-\emph{f(a, b)}\big) = 0
  \]

  \pause
  Rearranjando
  \[
    z
    = \emph{f(a, b)}
    + \emph{f_x(a, b)} \big(x-a\big)
    + \emph{f_y(a, b)} \big(y-b\big)
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Aproximação Linear}
  \emph{Aproximação linear},
  ou \emph{Linearização} de uma função diferenciável de duas variáveis \\
  em torno do ponto \((a, b)\)
  \pause
  \[
    L(x, y)
    = \emph{f(a, b)}
    + \emph{f_x(a, b)} (x-a)
    + \emph{f_y(a, b)} (y-b)
  \]
\end{frame}
