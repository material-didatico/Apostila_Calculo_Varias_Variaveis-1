\section{Plano Tangente à Superfície de Nível}

%------------------------------------------------------------------------------%
\begin{proofframe}{Justificativa}
  Primeiro vamos verificar que o gradiente é normal a superfície de nível

  \pause
  Considere uma função $f(x, y, z)$ diferenciável e sua superfície de nível
  \[
    f(x, y, z) = k
  \]
  \pause
  Assuma que a curva paramétrica
  \[
    r(t)
      = u(t)\veci + v(t)\vecj + w(t)\veck
      = \left(\begin{array}{c}
          u(t) \\ v(t) \\ w(t)
        \end{array}\right)
  \]
  \pause
  está contida na superfície de nível $k$ da função $f$
\end{proofframe}

%------------------------------------------------------------------------------%
\begin{proofframe}{Justificativa}
\begin{columns}
\column{0.5\linewidth}
  \begin{align*}
     \onslide<+->{f\big(u(t), v(t), w(t)\big) & = k \\[5mm]}
     \onslide<+->{\frac{d}{dt}f\big(u(t), v(t), w(t)\big) & = \frac{dk}{dt} \\[5mm]}
     \onslide<+->{\D{f}{x}\frac{du}{dt} + \D{f}{y}\frac{dv}{dt} + \D{f}{z}\frac{dw}{dt} & = 0}
  \end{align*}
\column{0.5\linewidth}
  \begin{align*}
    \onslide<+->{\Vetortri{\D{f}{x}}{\D{f}{y}}{\D{f}{z}}
                 & \cdot
                 \Vetortri{\frac{du}{dt}}{\frac{dv}{dt}}{\frac{dw}{dt}} = 0 \\[2mm]}
    \onslide<+->{\nabla f & \cdot \frac{dr}{dt} = 0}
  \end{align*}
\end{columns}
\end{proofframe}

%------------------------------------------------------------------------------%
\begin{proofframe}{Justificativa}
  Se tomarmos todas as curvas suaves que passam por um ponto \((a, b, c)\)

  \pause
  \vspace{\baselineskip}
  Seus vetores tangentes são todos ortogonais ao vetor gradiente no ponto

  \pause
  \vspace{\baselineskip}
  Todos os vetores tangentes estão no plano tangente a superfície
\end{proofframe}

%------------------------------------------------------------------------------%
\begin{frame}{Definições}
  O \emph{Plano Tangente} ao ponto \((a, b, c)\)
  \pause
  na superfície de nível
  \(f(x, y, z) = k\)

  \pause
  de uma função diferenciável $f$
  \pause
  é o plano passando pelo ponto \((a, b, c)\)

  \pause
  e normal a \(\nabla f (a, b, c)\)

  \pause
  \vspace{\baselineskip}
  A \emph{Reta Normal} da superfície no ponto é a reta passando pelo ponto

  \pause
  na direção do gradiente
  \(\nabla f (a, b, c)\)
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Equação do Plano Tangente}
  \[
    \emph{\nabla f(a, b, c)} \cdot (X-X_0) = 0
  \]
  \pause
  \[
    \emph{\nabla f(a, b, c)} \cdot \left(
        \begin{array}{c}
          x - a \\
          y - b \\
          z - c
        \end{array}
      \right) = 0
  \]

  \pause
  \[
    \emph{f_x(a, b, c)} (x-a) +
    \emph{f_y(a, b, c)} (y-b) +
    \emph{f_z(a, b, c)} (z-c) = 0
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Equação da Reta Normal}
  \[
    X = X_0 + t\,\emph{\nabla f(a, b, c)}
    \qquad\qquad
    t \in \R
  \]

  \pause
  \[
    \begin{pmatrix} x \\ y \\ z \end{pmatrix}
    = \begin{pmatrix} a\\ b \\ c \end{pmatrix}
    + t \begin{pmatrix}
      \emph{f_x(a, b, c)} \\
      \emph{f_y(a, b, c)} \\
      \emph{f_z(a, b, c)}
    \end{pmatrix}
  \]

  \pause
  \[
    \begin{cases}
      x = a   + t\;\emph{f_x(a, b, c)} \\
      y = b   + t\;\emph{f_y(a, b, c)} \\
      z\, = c + t\;\emph{f_z(a, b, c)}
    \end{cases}
  \]
\end{frame}

\section{Plano Tangente ao Gráfico de uma Função}

%------------------------------------------------------------------------------%
\begin{proofframe}{Justificativa}
  Queremos o Plano Tangente a superfície suave \(z=g(x, y)\) \\
  em um ponto \((a, b, c)\)
  onde \(c = g(a, b)\)

  \pause
  \vspace{\baselineskip}
  Podemos escrever \(z=g(x, y)\) como \(g(x, y) - z = 0\)

  \pause
  \vspace{\baselineskip}
  Queremos o plano tangente a curva de nível zero da função
  \[
    f(x, y, z) = g(x, y) - z = 0
  \]
\end{proofframe}

%------------------------------------------------------------------------------%
\begin{proofframe}{Justificativa}
  Derivadas de $F$
  \begin{align*}
    \onslide<+->{f_x(x, y, z)}\onslide<+->{ & = \D{}{x}\big(g(x, y) - z\big)}\onslide<+->{ = g_x(x, y) - 0}\onslide<+->{ = g_x(x, y)} \\[5mm]
    \onslide<+->{f_y(x, y, z)}\onslide<+->{ & = \D{}{y}\big(g(x, y) - z\big)}\onslide<+->{ = g_y(x, y) - 0}\onslide<+->{ = g_y(x, y)} \\[5mm]
    \onslide<+->{f_z(x, y, z)}\onslide<+->{ & = \D{}{z}\big(g(x, y) - z\big)}\onslide<+->{ =         0 - 1}\onslide<+->{ =  -1}
  \end{align*}
\end{proofframe}

%------------------------------------------------------------------------------%
\begin{proofframe}{Justificativa}
  Equação do Plano Tangente ao gráfico da função $f(x, y)$
  \[
    \emph{\nabla f(a, b, c)} \cdot \vetortri{x - a}{y - b}{z - c} = 0
  \]

  \pause
  \[
    \emph{f_x(a, b, c)} (x-a) +
    \emph{f_y(a, b, c)} (y-b) +
    \emph{f_z(a, b, c)} (z-c) = 0
  \]

  \pause
  \[
    {\color{magenta} g_x(a, b)} (x-a) +
    {\color{magenta} g_y(a, b)} (y-b) -
    (z-c) = 0
  \]
\end{proofframe}

%------------------------------------------------------------------------------%
\begin{frame}{Plano Tangente a superfície $z=f(x, y)$}
  O plano tangente ao gráfico de $f(x, y)$,
  \pause
  isto é, à superfície \(z=f(x, y)\)

  \pause
  de uma função diferenciável $f$
  \pause
  no ponto \((a, b)\) é

  \pause
  \[
    \D{f}{x}(a, b) \, \big(x-a\big) \;+\;
    \D{f}{y}(a, b) \, \big(y-b\big) \;-\;
    \big(z-f(a, b)\big) \;=\; 0
  \]
\end{frame}
