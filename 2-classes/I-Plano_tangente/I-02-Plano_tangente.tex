\section{Plano Tangente à Superfície de Nível}

%------------------------------------------------------------------------------%
\begin{proofframe}{Justificativa}
  Primeiro vamos verificar que o gradiente é normal a superfície de nível

  \pause
  Considere uma função $f(x, y)$ diferenciável e sua superfície de nível
  \[
    f(x, y, z) = k
  \]
  \pause
  Assuma que a curva paramétrica
  \[
    r(t)
      = g(t)\veci + h(t)\vecj + k(t)\veck
      = \left(\begin{array}{c}
          g(t) \\ h(t) \\ k(t)
        \end{array}\right)
  \]
  \pause
  está contida na superfície de nível $c$ da função $f$
\end{proofframe}

%------------------------------------------------------------------------------%
\begin{proofframe}{Justificativa}
\begin{columns}
\column{0.5\linewidth}
  \begin{align*}
     \onslide<+->{f\big(g(t), h(t), k(t)\big) & = k \\[5mm]}
     \onslide<+->{\frac{d}{dt}f\big(g(t), h(t), k(t)\big) & = \frac{dk}{dt} \\[5mm]}
     \onslide<+->{\D{f}{x}\frac{dg}{dt} + \D{f}{y}\frac{dh}{dt} + \D{f}{z}\frac{dk}{dt} & = 0}
  \end{align*}
\column{0.5\linewidth}
  \begin{align*}
    \onslide<+->{\begin{pmatrix}
        \D{f}{x} \\[3mm]
        \D{f}{y} \\[3mm]
        \D{f}{z}
      \end{pmatrix}
      \cdot
      \begin{pmatrix}
        \frac{dg}{dt} \\[3mm]
        \frac{dh}{dt} \\[3mm]
        \frac{dk}{dt}
      \end{pmatrix}
      & = 0 \\[2mm]}
    \onslide<+->{\nabla f \cdot \frac{dr}{dt} & = 0}
  \end{align*}
\end{columns}
\end{proofframe}

%------------------------------------------------------------------------------%
\begin{proofframe}{Justificativa}
  Se tomarmos todas as curvas suaves que passam por um ponto \((a, b)\)

  \pause
  \vspace{\baselineskip}
  Seus vetores tangentes são todos ortogonais ao vetor gradiente no ponto

  \pause
  \vspace{\baselineskip}
  Todos os vetores tangentes estão no plano tangente a superfície
\end{proofframe}

%------------------------------------------------------------------------------%
\begin{frame}{Definições}
  O \emph{Plano Tangente} ao ponto \((a, b, c)\)
  \pause
  na superfície de nível
  \(f(x, y, z) = k\)

  \pause
  de uma função diferenciável $f$
  \pause
  é o plano passando pelo ponto \((a, b, c)\)

  \pause
  e normal a \(\nabla f (a, b, c)\)

  \pause
  \vspace{\baselineskip}
  A \emph{Reta Normal} da superfície no ponto é a reta passando pelo ponto

  \pause
  na direção do gradiente
  \(\nabla f (a, b, c)\)
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Equação do Plano Tangente}
  \[
    \emph{\nabla f(a, b, c)} \cdot (X-X_0) = 0
  \]
  \pause
  \[
    \emph{\nabla f(a, b, c)} \cdot \left(
        \begin{array}{c}
          x - a \\
          y - b \\
          z - c
        \end{array}
      \right) = 0
  \]

  \pause
  \[
    \emph{f_x(a, b, c)} (x-a) +
    \emph{f_y(a, b, c)} (y-b) +
    \emph{f_z(a, b, c)} (z-c) = 0
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Equação da Reta Normal}
  \[
    X = X_0 + t\,\emph{\nabla f(a, b, c)}
    \qquad\qquad
    t \in \R
  \]

  \pause
  \[
    \begin{pmatrix} x \\ y \\ z \end{pmatrix}
    = \begin{pmatrix} a\\ b \\ c \end{pmatrix}
    + t \begin{pmatrix}
      \emph{f_x(a, b, c)} \\
      \emph{f_y(a, b, c)} \\
      \emph{f_z(a, b, c)}
    \end{pmatrix}
  \]

  \pause
  \[
    \begin{cases}
      x = a   + t\;\emph{f_x(a, b, c)} \\
      y = b   + t\;\emph{f_y(a, b, c)} \\
      z\, = c + t\;\emph{f_z(a, b, c)}
    \end{cases}
  \]
\end{frame}

\section{Plano Tangente ao Gráfico de uma Função}

%------------------------------------------------------------------------------%
\begin{proofframe}{Justificativa}
  Queremos o Plano Tangente a superfície suave \(z=f(x, y)\) \\
  em um ponto \((a, b, c)\)
  onde \(c = f(a, b)\)

  \pause
  \vspace{\baselineskip}
  Podemos escrever \(z=f(x, y)\) como \(f(x, y) - z = 0\)

  \pause
  \vspace{\baselineskip}
  Queremos o plano tangente a curva de nível zero da função
  \[
    F(x, y, z) = f(x, y) - z = 0
  \]
\end{proofframe}

%------------------------------------------------------------------------------%
\begin{proofframe}{Justificativa}
  Derivadas de $F$
  \begin{align*}
    \onslide<+->{F_x}\onslide<+->{ = \D{}{x}\big(f(x, y) - z\big)}\onslide<+->{ = f_x - 0}\onslide<+->{ = f_x} \\[5mm]
    \onslide<+->{F_y}\onslide<+->{ = \D{}{y}\big(f(x, y) - z\big)}\onslide<+->{ = f_y - 0}\onslide<+->{ = f_y} \\[5mm]
    \onslide<+->{F_z}\onslide<+->{ = \D{}{z}\big(f(x, y) - z\big)}\onslide<+->{ =   0 - 1}\onslide<+->{ =  -1}
  \end{align*}
\end{proofframe}

%------------------------------------------------------------------------------%
\begin{proofframe}{Justificativa}
  Equação do Plano Tangente ao gráfico da função $f(x, y)$
  \[
    \emph{\nabla F(a, b, c)} \cdot
    \begin{pmatrix}
      x - a \\
      y - b \\
      z - c
    \end{pmatrix} = 0
  \]

  \pause
  \[
    \emph{F_x(a, b, c)} (x-a) +
    \emph{F_y(a, b, c)} (y-b) +
    \emph{F_z(a, b, c)} (z-c) = 0
  \]

  \pause
  \[
    {\color{magenta} f_x(a, b)} (x-a) +
    {\color{magenta} f_y(a, b)} (y-b) -
    (z-c) = 0
  \]
\end{proofframe}

%------------------------------------------------------------------------------%
\begin{frame}{Plano Tangente a superfície $z=f(x, y)$}
  O plano tangente à superfície \(z=f(x, y)\)

  de uma função diferenciável $f$
  no ponto \(\big(a,\, b,\, f(a, b)\big)\) é

  \[
    f_x(a, b) (x-a) \;+\;
    f_y(a, b) (y-b) \;-\;
    (z-c) \;=\; 0
  \]
\end{frame}
