%------------------------------------------------------------------------------%
\begin{question}
Utilize a linearização da função
\[
  f(x, y) = e^{x^2 - y^2} \cos(xy)
\]
no ponto \(\left(\sqrt{\pi}, \sqrt{\pi}\right)\),
para obter uma estimativa para \(f(2, 1)\)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Avaliando a derivada parcial em $x$}
  \begin{align*}
    \onslide<+->{\D{f}{x}}
    \onslide<+->{& = \D{}{x}\left(e^{x^2 - y^2} \cos(xy)\right)   \\[2mm]}
    \onslide<+->{& = \D{}{x}\left(e^{x^2 - y^2}\right)  \cos(xy)
                   + e^{x^2 - y^2} \D{}{x}\left(\cos(xy)\right)   \\[2mm]}
    \onslide<+->{& = e^{x^2 - y^2} \D{}{x}\left(x^2 - y^2\right)  \cos(xy)
                   - e^{x^2 - y^2} \sin(xy)\D{}{x}\left(xy\right) \\[2mm]}
    \onslide<+->{& = e^{x^2 - y^2} 2x \cos(xy)
                   - e^{x^2 - y^2} \sin(xy)y                      \\[2mm]}
    \onslide<+->{& = e^{x^2 - y^2} \big(2x \cos(xy) - y \sin(xy)\big)}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Avaliando a derivada parcial em $x$}
  \begin{align*}
    \onslide<+->{\D{f}{x}\left(\sqrt{\pi}, \sqrt{\pi}\right)}
    \onslide<+->{& = e^{\sqrt{\pi}^2 - \sqrt{\pi}^2}
      \Big(2\sqrt{\pi} \cos\left(\sqrt{\pi}\sqrt{\pi}\right)
          - \sqrt{\pi} \sin\left(\sqrt{\pi}\sqrt{\pi}\right)\Big) \\[2mm]}
    \onslide<+->{& = e^{0}
      \Big(2\sqrt{\pi} \cos\left(\pi\right)
          - \sqrt{\pi} \sin\left(\pi\right)\Big) \\[2mm]}
    \onslide<+->{& = 2\sqrt{\pi} (-1) - \sqrt{\pi} 0 \\[2mm]}
    \onslide<+->{& = -2\sqrt{\pi}}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Avaliando a derivada parcial em $y$}
  \begin{align*}
    \onslide<+->{\D{f}{y}}
    \onslide<+->{& = \D{}{y}\left(e^{x^2 - y^2} \cos(xy)\right)   \\[2mm]}
    \onslide<+->{& = \D{}{y}\left(e^{x^2 - y^2}\right)  \cos(xy)
                   + e^{x^2 - y^2} \D{}{y}\left(\cos(xy)\right)   \\[2mm]}
    \onslide<+->{& = e^{x^2 - y^2} \D{}{y}\left(x^2 - y^2\right)  \cos(xy)
                   - e^{x^2 - y^2} \sin(xy)\D{}{y}\left(xy\right) \\[2mm]}
    \onslide<+->{& = e^{x^2 - y^2} \left(-2y\right) \cos(xy)
                   - e^{x^2 - y^2} \sin(xy)x                      \\[2mm]}
    \onslide<+->{& = - e^{x^2 - y^2} \big(2y \cos(xy) + x \sin(xy)\big)}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Avaliando a derivada parcial em $y$}
  \begin{align*}
    \onslide<+->{\D{f}{y}\left(\sqrt{\pi}, \sqrt{\pi}\right)}
    \onslide<+->{& = - e^{\sqrt{\pi}^2 - \sqrt{\pi}^2}
      \Big(2\sqrt{\pi} \cos(\sqrt{\pi}\sqrt{\pi})
          + \sqrt{\pi} \sin(\sqrt{\pi}\sqrt{\pi})\Big) \\}
    \onslide<+->{& = - e^{0} \Big(2\sqrt{\pi} (-1) + \sqrt{\pi} 0\Big) \\}
    \onslide<+->{& = 2\sqrt{\pi}}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Avaliando a função no ponto}\
  \begin{align*}
    \onslide<+->{f(\sqrt{\pi}, \sqrt{\pi})}
    \onslide<+->{& = e^{x^2 - y^2} \cos(xy) \\}
    \onslide<+->{& = e^{\sqrt{\pi}^2 - \sqrt{\pi}^2} \cos\left(\sqrt{\pi}\sqrt{\pi}\right) \\}
    \onslide<+->{& = e^{0} \cos\left(\pi\right) \\}
    \onslide<+->{& = -1}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Aproximação linear}
  \begin{align*}
    \onslide<+->{L(x, y)}
    \onslide<+->{& = f(a, b) + f_x(a, b)(x-a) + f_y(a, b)(y-b) \\}
    \onslide<+->{& = f\left(\sqrt{\pi}, \sqrt{\pi}\right)
      + f_x\left(\sqrt{\pi}, \sqrt{\pi}\right)\left(x-\sqrt{\pi}\right)
      + f_y\left(\sqrt{\pi}, \sqrt{\pi}\right)\left(y-\sqrt{\pi}\right) \\}
    \onslide<+->{& = -1
      - 2\sqrt{\pi}\left(x-\sqrt{\pi}\right)
      + 2\sqrt{\pi}\left(y-\sqrt{\pi}\right)  \\}
    \onslide<+->{& = -1 - 2\sqrt{\pi}x + 2\pi + 2\sqrt{\pi}y - 2\pi \\}
    \onslide<+->{& = 2\sqrt{\pi}(y-x) -1}
  \end{align*}
\end{resolution}
