\begin{question}
  Encontre o plano tangente e a reta normal da superfície
  \[
    f(x, y, z) = x^2 + y^2 + z - 9 = 0
  \]
  no ponto \((1, 2, 4)\)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
\begin{columns}
\column{0.45\linewidth}
  \begin{align*}
    \onslide<+->{\D{f}{x} & = \D{}{x} \left(x^2 + y^2 + z - 9\right) = 2x \\[2mm]}
    \onslide<+->{\D{f}{y} & = \D{}{y} \left(x^2 + y^2 + z - 9\right) = 2y \\[2mm]}
    \onslide<+->{\D{f}{z} & = \D{}{z} \left(x^2 + y^2 + z - 9\right) = 1}
  \end{align*}
\column{0.55\linewidth}
  \[
    \onslide<+->{\nabla f}
    \onslide<+->{= 2x\veci + 2y\vecj + \veck}
    \onslide<+->{= \left(\begin{array}{c}
        2x \\ 2y \\ 1
      \end{array}\right)}
  \]
  \[
    \onslide<+->{\nabla f(1, 2, 4)}
    \onslide<+->{= \left(2x\veci + 2y\vecj + \veck\right)\evalat{(1, 2, 4)}{}}
  \]
  \[
    \hphantom{\nabla f(1, 2, 4)}
    \onslide<+->{= 2\veci + 4\vecj + \veck}
    \onslide<+->{= \left(\begin{array}{c}
         2 \\ 4 \\ 1
      \end{array}\right)}
  \]
\end{columns}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  Plano tangente
  \[
    \pause
    \nabla f(a, b, c) \cdot  \left(
    \begin{array}{c}
      x - a \\
      y - b \\
      z - c
    \end{array}
    \right) = 0
  \]

  \[
    \pause
    f_x(a, b, c) (x-a) +
    f_y(a, b, c) (y-b) +
    f_z(a, b, c) (z-c) = 0
  \]

  \[
    \pause
    f_x(1, 2, 4) (x-1) +
    f_y(1, 2, 4) (y-2) +
    f_z(1, 2, 4) (z-4) = 0
  \]

  \[
    \pause
    2 (x-1) + 4 (y-2) + (z-4) = 0
  \]

  \[
    \pause
    2x + 4y + z = 14
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  Reta normal
  \pause
  \[
    X = X_0 + t\,\nabla f(a, b, c)
    \qquad\qquad
    t \in \R
  \]

  \pause
  \[
    \begin{cases}
      x = a + t\,f_x(a, b, c) \\
      y = b + t\,f_y(a, b, c) \\
      z = c + t\,f_z(a, b, c)
    \end{cases}
  \]

  \pause
  \[
    \begin{cases}
      x = 1 + 2t \\
      y = 2 + 4t \\
      z = 4 + t
    \end{cases}
  \]
\end{resolution}
