%------------------------------------------------------------------------------%
\begin{question}
  Encontre a linearização de
  \[
    f(x, y) = x^2 - xy + \frac{y^2}{2} + 3
  \]
  no ponto \((3, 2)\)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Avaliando a Derivada Parcial em $x$}
  \begin{align*}
    \onslide<+->{f_x(x, y)}
    \onslide<+->{& = \D{}{x} \left(x^2 - xy + \frac{y^2}{2} + 3\right)}
    \onslide<+->{= 2x - y
      \\[3mm]}
    \onslide<+->{f_x(3, 2)}
    \onslide<+->{& = \left(2x - y\right)\evalat{(3, 2)}{}}
    \onslide<+->{= 2\times 3 - 2}
    \onslide<+->{= 4}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Avaliando a Derivada Parcial em $y$}
  \begin{align*}
    \onslide<+->{f_x(x, y)}
    \onslide<+->{& = \D{}{x} \left(x^2 - xy + \frac{y^2}{2} + 3\right)}
    \onslide<+->{= 2x - y
      \\[3mm]}
    \onslide<+->{f_y(3, 2)}
    \onslide<+->{& = \left(-x + y\right)\evalat{(3, 2)}{}}
    \onslide<+->{= -3 + 2}
    \onslide<+->{= -1}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Avaliando a Função}
\begin{align*}
  \onslide<+->{f(3, 2)}
  \onslide<+->{& = \left(x^2 - xy + \frac{y^2}{2} + 3\right)\evalat{(3, 2)}{}}
  \onslide<+->{= 3^2 - 3\times 2 + \frac{2^2}{2} + 3}
  \onslide<+->{= 8                                                   \\}
\end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Aproximação linear}
  \begin{align*}
    \onslide<+->{L(x, y)}
    \onslide<+->{& = f(a, b) + f_x(a, b)(x-a) + f_y(a, b)(y-b) \\}
    \onslide<+->{& = f(3, 2) + f_x(3, 2)(x-3) + f_y(3, 2)(y-2)                 \\}
    \onslide<+->{& = 8 + 4(x-3) + (-1)(y-2)                                    \\}
    \onslide<+->{& = 4x -y -2}
  \end{align*}
\end{resolution}
