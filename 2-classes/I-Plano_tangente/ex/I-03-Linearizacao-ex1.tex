%------------------------------------------------------------------------------%
\begin{question}
Encontre linearização da função
\[
  f(x, y) = x^2 - xy - y^2
\]
no ponto \((1, 1)\)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Linearização de $f$ no ponto \((1, 1)\)}
  \begin{align*}
    \onslide<+->{L(x, y)}
    \onslide<+->{& = f(a, b) + f_x(a, b)(x-a) + f_y(a, b)(y-b) \\[2mm]}
    \onslide<+->{& = f(1, 1) + f_x(1, 1)(x-1) + f_y(1, 1)(y-1)}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Avaliando a derivada parcial em $x$}
  \[
    f_x(x, y)
    \pause
    \;=\; \D{f}{x}
    \pause
    \;=\; \D{}{x}\left[x^2 - xy - y^2\right]
    \pause
    \;=\; 2x -y
  \]

  \pause
  \bigskip
  \[
    f_x(1, 1)
    \pause
    \;=\; \left(2x -y\right)\evalat{(1, 1)}{}
    \pause
    \;=\; 2\times 1 - 1
    \pause
    \;=\; 1
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Avaliando a derivada parcial em $y$}
  \[
    f_y(x, y)
    \pause
    \;=\; \D{f}{y}
    \pause
    \;=\; \D{}{y}\left[x^2 - xy - y^2\right]
    \pause
    \;=\; -x -2y
  \]

  \pause
  \bigskip
  \[
    f_y(1, 1)
    \pause
    \;=\; \left(-x -2y\right)\evalat{(1, 1)}{}
    \pause
    \;=\; -1 -2 \times 1
    \pause
    \;=\; -3
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Avaliando a função}
  \[
    f(1, 1)
    \pause
    \;=\; \left(x^2 - xy - y^2\right)\evalat{(1, 1)}{}
    \pause
    \;=\; 1^2 - 1\times 1 - 1^2
    \pause
    \;=\; -1
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Linearização}
  \begin{align*}
    \onslide<+->{L(x, y)}
    \onslide<+->{& = f(1, 1) + f_x(1, 1)(x-1) + f_y(1, 1)(y-1) \\[2mm]}
    \onslide<+->{& = -1 + 1(x-1) -3(y-1) \\[2mm]}
    \onslide<+->{& = -1 + x -1 -3y +3 \\[2mm]}
    \onslide<+->{& = x - 3y + 1}
  \end{align*}
\end{resolution}
