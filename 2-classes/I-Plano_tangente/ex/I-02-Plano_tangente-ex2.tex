%------------------------------------------------------------------------------%
\begin{question}
  Encontre o plano tangente ao gráfico da função
  \[
    f(x, y) = x\cos(y) - ye^x
  \]
  no ponto \((0, 0)\)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Avaliando o valor da função no ponto}
  \[
    \pause
    c
    \pause
    \;=\; f(0, 0)
    \pause
    \;=\; \left(x\cos(y) - ye^x\right)\evalat{(0, 0)}{}
    \pause
    \;=\; 0\times\cos(0) - 0\times e^0
    \pause
    \;=\; 0
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Derivadas parciais de $f(x, y)$}
  \begin{align*}
    \onslide<+->{\D{f}{x}(x, y)}
    \onslide<+->{& = \D{}{x} \left(x\cos(y) - ye^x\right)}
    \onslide<+->{  = \cos(y) - ye^x                        \\[1mm]}
    \onslide<+->{\D{f}{y}(x, y)}
    \onslide<+->{& = \D{}{y} \left(x\cos(y) - ye^x\right)}
    \onslide<+->{  = -x\sin(y) - e^x}
  \end{align*}
  \begin{align*}
    \onslide<+->{\D{f}{x}(0, 0)}
    \onslide<+->{& = \left(  \cos(y) - ye^x\right) \evalat{(0, 0)}{}}
    \onslide<+->{  = 1                                    \\[1mm]}
    \onslide<+->{\D{f}{y}(0, 0)}
    \onslide<+->{& = \left(-x\sin(y) - e^x \right) \evalat{(0, 0)}{}}
    \onslide<+->{  = -1}
  \end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  Plano tangente
  \[
    \emph{f_x(a, b)} \big(x-a\big) +
    \emph{f_y(a, b)} \big(y-b\big) -
    \big(z-\emph{f(a, b)}\big) = 0
  \]

  \pause
  \[
    \emph{f_x(0, 0)} \big(x-0\big) +
    \emph{f_y(0, 0)} \big(y-0\big) -
    \big(z-\emph{f(0, 0)}\big) = 0
  \]

  \pause
  \[
    \emph{1} (x-0) -
    \emph{1} (y-0) -
    (z-\emph{0}) = 0
  \]

  \pause
  \[
    x - y - z = 0
  \]
\end{resolution}
