\section{Definição Algébrica}

\begin{frame}{Cônicas e Quádricas}
  \emph{Cônica}:
  pontos no plano, $\R^2$, que são solução de uma equação de segundo grau
  \[
    Ax^{\emph{2}} + By^{\emph{2}} + Cxy + Dx + Ey + F = 0
  \]

  \pause
  \emph{Quádrica}:
  pontos no espaço, $\R^3$, que são solução de uma equação de segundo grau
  \[
    Ax^{\emph{2}} + By^{\emph{2}} + Cz^{\emph{2}} +
    Dxy  + Exz  + Fyz +
    Gx   + Hy   + Iz + G = 0
  \]
\end{frame}

\section{Equações}

%------------------------------------------------------------------------------%
\begin{frame}{Parábola}
  \begin{columns}
    \column{0.4\linewidth}
    Pontos equidistantes de um ponto fixo (foco)
    e uma reta dada (diretriz)
    \[
      y = \frac{x^2}{4p}
    \]
    \column{0.6\linewidth}
    \centering
    \includegraphics{parabola}
  \end{columns}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Elipse}
  \begin{columns}
    \column{0.4\linewidth}
    Pontos cuja soma das distâncias até
    os focos é constante
    \[
      \frac{x^2}{a^2} + \frac{y^2}{b^2} = 1
    \]
    \column{0.6\linewidth}
    \centering
    \includegraphics{elipse}
  \end{columns}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Hipérbole}
  \begin{columns}
    \column{0.4\linewidth}
    Pontos cujas distâncias a partir de dois pontos fixos (focos)
    tem diferença constante
    \[
      \frac{x^2}{a^2} - \frac{y^2}{b^2} = 1
    \]
    A hipérbole tem duas assintotas
    \column{0.6\linewidth}
    \centering
    \includegraphics{hiperbole}
  \end{columns}
\end{frame}
