\section{Cônicas}

\begin{frame}{Cônicas e Quádricas}
  \emph{Cônica}:
  Gráfico de uma equação de segundo grau no plano
  \[
    Ax^{\emph{2}} + By^{\emph{2}} + Cxy + Dx + Ey + F = 0
  \]

  \pause
  \emph{Quádrica}:
  gráfico de uma equação de segundo grau em $\R^3$
  \[
    Ax^{\emph{2}} + By^{\emph{2}} + Cz^{\emph{2}} +
    Dxy  + Exz  + Fyz +
    Gx   + Hy   + Iz + G = 0
  \]
\end{frame}

\begin{frame}{Parábola}

    \]
    \column{0.6\linewidth}
    \centering
    \includegraphics[width=\linewidth]{parabola}
  \end{columns}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Elipse}
  \begin{columns}
    \column{0.4\linewidth}
    Pontos cuja soma das distâncias até
    os focos é constante
    \[
      \frac{x^2}{a^2} + \frac{y^2}{b^2} = 1
    \]
    \column{0.6\linewidth}
    \centering
    \includegraphics[width=0.8\linewidth]{elipse}
  \end{columns}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Hipérbole}
  \begin{columns}
    \column{0.4\linewidth}
    Pontos cujas distâncias a partir de dois pontos fixos (focos)
    tem diferença constante
    \[
      \frac{x^2}{a^2} - \frac{y^2}{b^2} = 1
    \]
    A hipérbole tem duas assintotas
    \column{0.6\linewidth}
    \centering
    \includegraphics[width=0.8\linewidth]{hiperbole}
  \end{columns}
\end{frame}
