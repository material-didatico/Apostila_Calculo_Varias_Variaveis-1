\section{Quádricas}

%------------------------------------------------------------------------------%
\begin{frame}{Cônicas e Quádricas}
  \emph{Cônica}:
  pontos no plano, $\R^2$, que são solução de uma equação de segundo grau
  \[
    Ax^{\emph{2}} + By^{\emph{2}} + Cxy + Dx + Ey + F = 0
  \]

  \pause
  \emph{Quádrica}:
  pontos no espaço, $\R^3$, que são solução de uma equação de segundo grau
  \[
    Ax^{\emph{2}} + By^{\emph{2}} + Cz^{\emph{2}} +
    Dxy  + Exz  + Fyz +
    Gx   + Hy   + Iz + G = 0
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Quádricas}
  \begin{itemize}
    \item Elipsoides
    \item Paraboloides
    \item Cones elípticos
    \item Hiperboloides
  \end{itemize}
\end{frame}

\begin{frame}{Elipsoides}
  \[
    \frac{x^2}{a^2} +
    \frac{y^2}{b^2} +
    \frac{z^2}{c^2} = 1
  \]
\end{frame}
