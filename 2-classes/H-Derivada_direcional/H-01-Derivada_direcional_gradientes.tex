\section{Derivada Direcional}

%------------------------------------------------------------------------------%
\framedgraphic{Derivada Direcional}{derivada_direcional}{}

%------------------------------------------------------------------------------%
\begin{frame}{Definição}
  A derivada de $f$ em $(a, b)$
  \pause
  na direção do vetor unitário
  \[
    u
    = u_1 \veci + u_2 \vecj
    = \vetor{u_1}{u_2}
    \qquad
    \qquad
    \norm{u} = 1
  \]
  \pause
  é o número
  \[
    D_u f(a, b)
    = \lim_{h\to 0} \frac{\;f( a + hu_1, b + hu_2)-f(a, b)\;}{h}
  \]
\end{frame}

%------------------------------------------------------------------------------%
\framedgraphic{Interpretação da Derivada Direcional}{derivada_direcional_interpretacao}{}

%------------------------------------------------------------------------------%
\begin{frame}{Interpretação da Derivada Direcional}
  A derivada direcional é a taxa de variação da função $f$ na direção do vetor $u$

  \pause
  \vspace{\baselineskip}
  Se \(u=\veci=\vetor{1}{0}\)
  \pause
  então
  \(
    D_u f = \D{f}{x}
  \)

  \pause
  \vspace{\baselineskip}
  Se \(u=\vecj=\vetor{0}{1}\)
  \pause
  então
  \(
    D_u f = \D{f}{y}
  \)
\end{frame}

\section{Vetor Gradiente}

%------------------------------------------------------------------------------%
\begin{frame}{Vetor Gradiente}
  O \emph{vetor gradiente} de $f(x, y)$ é

  \[
    \pause
    \nabla f
    \pause
    \;=\; \D{f}{x}\veci \,+\, \D{f}{y}\vecj
    \pause
    \;=\; \left(\begin{array}{c}
            \displaystyle\D{f}{x} \\[5mm]
            \displaystyle\D{f}{y}
          \end{array}\right)
  \]

  \pause
  O símbolo \(\nabla\) é chamado \emph{nabla}
\end{frame}

\section{Calculando a Derivada Direcional}

%------------------------------------------------------------------------------%
\begin{frame}{Calculando a Derivada Direcional}
  Considere a reta que passa por \((a, b)\)
  \pause
  na direção de \(u=u_1\veci+u_2\vecj\)
  \pause
  \[
    x = a + su_1 \qquad
    y = b + su_2
  \]

  Restringindo $f$ à reta
  \[
    f(s) = f\big(x(s), y(s)\big)
  \]
\end{frame}

\begin{frame}{Calculando a Derivada Direcional}
\begin{align*}
  \onslide<+->{D_u f }
  \onslide<+->{& = \frac{df}{ds}                                 \\[2mm]}
  \onslide<+->{& = \D{f}{x}\frac{dx}{ds} + \D{f}{y}\frac{dy}{ds} \\[2mm]}
  \onslide<+->{& = \D{f}{x}u_1 + \D{f}{y}u_2                     \\[2mm]}
  \onslide<+->{& = \left(\begin{array}{c} F_x \\ F_y \end{array}\right)
                   \cdot \vetor{u_1}{u_2}                        \\[2mm]}
  \onslide<+->{& = \emph{\nabla f \cdot u}}
\end{align*}
\end{frame}
