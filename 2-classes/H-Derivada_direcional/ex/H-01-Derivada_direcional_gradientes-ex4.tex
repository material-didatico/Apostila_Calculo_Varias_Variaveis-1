%------------------------------------------------------------------------------%
\begin{question}
  Encontre a derivada de
  \(
    f(x, y) = \ln\left(x^2 + y^2 + 1\right)
  \)
  no ponto
  \(
    (1, 2)
  \) \\

  \smallskip
  na direção
  \(
    v = \vetor{-2}{1}
  \)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  O vetor $v$ não é unitário,
  \pause
  então precisamos dividi-lo pelo seu
  comprimento
  \[
    \pause
    u
    \;=\; \frac{v}{\abs{v}}
    \pause
    \;=\; \frac{1}{\sqrt{(-2)^2 + 1^2}} \vetor{-2}{1}
    \pause
    \;=\; \frac{1}{\sqrt{5}} \vetor{-2}{1}
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Calculando a derivada parcial em $x$}
  \[
    \D{f}{x}
    \pause
    \;=\; \D{}{x} \ln\left(x^2 + y^2 + 1\right)
    \pause
    \;=\; \frac{1}{x^2 + y^2 + 1} \D{}{x}\left(x^2 + y^2 + 1\right)
    \pause
    \;=\; \frac{2x}{x^2 + y^2 + 1}
  \]

  \pause
  \bigskip
  \[
    \D{f}{x}(1, 2)
    \pause
    \;=\; \frac{2x}{x^2 + y^2 + 1}\evalat{(1, 2)}{}
    \pause
    \;=\; \frac{2\times 1}{1^2 + 2^2 + 1}
    \pause
    \;=\; \frac{2}{6}
    \pause
    \;=\; \frac{1}{3}
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Calculando a derivada parcial em $y$}
  \[
    \D{f}{y}
    \pause
    \;=\; \D{}{y} \ln\left(x^2 + y^2 + 1\right)
    \pause
    \;=\; \frac{1}{x^2 + y^2 + 1} \D{}{y}\left(x^2 + y^2 + 1\right)
    \pause
    \;=\; \frac{2y}{x^2 + y^2 + 1}
  \]

  \pause
  \bigskip
  \[
    \D{f}{y}(1, 2)
    \pause
    \;=\; \frac{2y}{x^2 + y^2 + 1}\evalat{(1, 2)}{}
    \pause
    \;=\; \frac{2\times 2}{1^2 + 2^2 + 1}
    \pause
    \;=\; \frac{4}{6}
    \pause
    \;=\; \frac{2}{3}
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Gradiente no ponto $(1, 2)$}
  \[
    \nabla f(x, y)
    \;=\; \Vetor{\frac{2x}{x^2 + y^2 + 1}}{\frac{2y}{x^2 + y^2 + 1}}
    \pause
    \qquad\qquad\qquad
    \nabla f(1, 2)
    \;=\; \Vetor{\frac{1}{3}}{\frac{2}{3}}
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Calculando a Derivada Direcional}
  \[
    D_u f(1, 2)
    \pause
    \;=\; \nabla f(1, 2) \cdot u
    \pause
    \;=\; \Vetor{\frac{1}{3}}{\frac{2}{3}} \cdot
          \Vetor{\frac{-2}{\sqrt{5}}}{\frac{1}{\sqrt{5}}}
    \pause
    \;=\; \frac{-2}{3\sqrt{5}} + \frac{2}{3\sqrt{5}}
    \pause
    \;=\; 0
  \]
\end{resolution}
