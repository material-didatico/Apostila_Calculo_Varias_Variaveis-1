%------------------------------------------------------------------------------%
\begin{question}
  Encontre a derivada de
  \(
    f(x, y) = xe^y + \cos(xy)
  \)
  no ponto
  \(
    (2, 0)
  \) \\

  \smallskip
  na direção
  \(
    v=3\veci-4\vecj
  \)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  O vetor $v$ não é unitário,
  \pause
  então precisamos dividi-lo pelo seu
  comprimento
  \[
    \pause
    u
    \;=\; \frac{v}{\abs{v}}
    \pause
    \;=\; \frac{3\veci-4\vecj}{\sqrt{3^2+(-4)^2}}
    \pause
    \;=\; \frac{3\veci-4\vecj}{\sqrt{9+16}}
    \pause
    \;=\; \frac{3\veci-4\vecj}{\sqrt{25}}
    \pause
    \;=\; \frac{3}{5}\veci - \frac{4}{5}\vecj
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Calculando a derivada parcial em $x$}
  \[
    \D{f}{x}
    \pause
    \;=\;  \D{}{x}\left(xe^y + \cos(xy)\right)
    \pause
    \;=\;  e^y - y\sin(xy)
  \]

  \pause
  \bigskip
  \[
    \D{f}{x}(2, 0)
    \pause
    \;=\;  e^0 - 0\sin(2\times 0)
    \pause
    \;=\;  1
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Calculando a derivada parcial em $y$}
  \[
    \D{f}{y}
    \pause
    \;=\; \D{}{y}\left(xe^y + \cos(xy)\right)
    \pause
    \;=\; xe^y - x\sin(xy)
  \]

  \pause
  \bigskip
  \[
    \D{f}{y}(2, 0)
    \pause
    \;=\; 2e^0 - 2\sin(2\times 0)
    \pause
    \;=\; 2
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Gradiente no ponto $(2, 0)$}
  \[
    \nabla f(x, y)
    \;=\; \Vetor{e^y - y\sin(xy)}{xe^y - x\sin(xy)}
    \pause
    \qquad\qquad
    \nabla f(2, 0)
    = \vetor{1}{2}
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Calculando a Derivada Direcional}
  \[
    D_u f(2, 0)
    \pause
    \;=\; \nabla f(2, 0) \cdot u
    \pause
    \;=\; \vetor{1}{2} \cdot \vetor{\sfrac{3}{5}}{-\sfrac{4}{5}}
  \]

  \[
    \hphantom{D_u f(2, 0){}}
    \pause
    \;=\; 1 \frac{3}{5} + 2 \frac{-4}{5}
    \pause
    \;=\; \frac{3}{5} - \frac{8}{5}
    \pause
    \;=\; - \frac{5}{5}
    \pause
    \;=\; -1
  \]
\end{resolution}
