\section{Vetor Tangente}

%------------------------------------------------------------------------------%
\begin{frame}{Curva Paramétrica}
  $x$ e $y$ são dados como funções
  \[
    x = f(t)     \qquad
    y = g(t)     \qquad
    t \in I
  \]

  \pause
  Vetorialmente escrevemos
  \[
    \begin{bmatrix}  x    \\ y    \end{bmatrix} =
    \begin{bmatrix}  f(t) \\ g(t) \end{bmatrix}
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Vetor Tangente}
  Se $f$ e $g$ forem deriváveis em $t$, o \emph{vetor tangente} é
  \pause
  \[
    \frac{d}{dt}\begin{bmatrix}  x    \\ y    \end{bmatrix} =
    \pause
    \begin{bmatrix}  x    \\ y    \end{bmatrix}' =
    \pause
    \begin{bmatrix} \;\dfrac{df}{dt}\;\null \\[5mm] \;\dfrac{dg}{dt}\;\null \end{bmatrix} =
    \pause
    \begin{bmatrix} \,f'(t)\,\null \\[2mm] \,g'(t)\,\null \end{bmatrix}
  \]
\end{frame}
