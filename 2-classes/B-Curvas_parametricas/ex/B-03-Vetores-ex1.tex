\begin{question}
  Avalie e esboce o vetor
  \[
    v(x, y) = \vetor{2x+y-1}{1+y-x}
  \]
  nos pontos
  \(( 0,  0)\),
  \(( 1,  1)\),
  \(( 1, -1)\)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Ponto \((0, 0)\)}
\begin{columns}
\column{0.5\linewidth}
  \begin{align*}
    \onslide<+->{v(0, 0)}
    \onslide<+->{& = \vetor{2x+y-1}{1+y-x} \evalat{(0, 0)}{} \\}
    \onslide<+->{& = \vetor{2\times 0+0-1}{1+0-0} \\}
    \onslide<+->{& = \vetor{-1}{1}}
  \end{align*}
\column{0.5\linewidth}
  \onslide<+->{\includegraphics{B_03_exemplo_1_vetor_1}}
\end{columns}
\end{resolution}
