\begin{question}
  Descreva e esboce as curvas de nível da função \(f(x, y) = xy\)
\end{question}

\begin{resolution}{Solução}
  \onslide<+->{Para cada valor $c \neq 0$}

  \begin{align*}
    \onslide<+->{f(x, y) & = c \\}
    \onslide<+->{xy & = c \\}
    \onslide<+->{y & = \frac{c}{x}}
  \end{align*}

  \onslide<+->{Para $c=0$, ou $x=0$ ou $y=0$}
\end{resolution}
