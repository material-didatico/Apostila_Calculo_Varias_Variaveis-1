%------------------------------------------------------------------------------%
\begin{question}
  Encontre as funções $f_x$ e $f_y$ sendo que
  \(
    f(x, y) = \frac{2y}{y+\cos(x)}
  \)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
\begin{align*}
  \onslide<+->{\D{f}{x}}
  \onslide<+->{& = \D{}{x} \left(\frac{2y}{y+\cos(x)}\right)} \\
  \onslide<+->{& = \D{}{x} \Big[ 2y \big( y+\cos(x)\big)^{-1} \Big]} \\
  \onslide<+->{& = 2y \D{}{x} \big( y + \cos(x) \big)^{-1}} \\
  \onslide<+->{& = 2y (-1)\big( y + \cos(x) \big)^{-2} \; \D{}{x}\left(y+\cos(x)\right)}\\
  \onslide<+->{& = -2y\big( y + \cos(x) \big)^{-2} \; \big(-\sin(x)\big)}\\
  \onslide<+->{& = \frac{2y \sin(x)\;}{\left(y+\cos(x)\right)^2}}
\end{align*}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
\begin{align*}
  \onslide<+->{\D{f}{y}}
  \onslide<+->{& = \;\D{}{y} \left(\frac{2y}{y+\cos(x)}\right)} \\
  \onslide<+->{& = 2\;\dfrac{\displaystyle\;\D{y}{y}\left(y+\cos(x)\right) - y \D{}{y}\left(y+\cos(x)\right)\;}
                            {\left(y+\cos(x)\right)^2}} \\
  \onslide<+->{& = 2\;\dfrac{\;\left(y +\cos(x)\right) - y(1 + 0)\;}
                            {\left(y+\cos(x)\right)^2}} \\
  \onslide<+->{& = \frac{2\cos(x)}{\,\left(y+\cos(x)\right)^2\,}}
\end{align*}
\end{resolution}
