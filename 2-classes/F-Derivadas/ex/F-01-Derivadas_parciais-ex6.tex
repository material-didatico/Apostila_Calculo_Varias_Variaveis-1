%------------------------------------------------------------------------------%
\begin{question}
  Calcule as derivadas parciais de primeira ordem da função
  \[
    f(x,y) = \arctan\left(\frac{x}{y}\right)
  \]

  \pause
  \[
    \frac{d}{du} \arctan(u) = \arctan'(u) = \frac{1}{1 + u^2}
  \]
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
\begin{columns}
\column{0.5\linewidth}
  \begin{align*}
    \onslide<+->{\D{f}{x}}
    \onslide<+->{& = \D{}{x}\left(\arctan\left(\frac{x}{y}\right)\right)} \\
    \onslide<+->{& = \arctan'\left(\frac{x}{y}\right) \D{}{x}\left(\frac{x}{y}\right)} \\
    \onslide<+->{& = \frac{1}{1+\left(\frac{x}{y}\right)^2} \D{}{x}\left(\frac{x}{y}\right)} \\
    \onslide<+->{& = \frac{1}{1+\left(\frac{x}{y}\right)^2}\left(\frac{1}{y}\right)}
  \end{align*}
\column{0.5\linewidth}
\begin{align*}
  \onslide<+->{\D{f}{x}}
  \onslide<+->{& = \frac{1}{1+\frac{x^2}{y^2}}\left(\frac{1}{y}\right)} \\
  \onslide<+->{& = \frac{1}{\;\dfrac{y^2+x^2}{y^2}\;}\left(\frac{1}{y}\right)} \\
  \onslide<+->{& = \frac{y^2}{y^2+x^2}\left(\frac{1}{y}\right)} \\
  \onslide<+->{& = \frac{y}{x^2 + y^2}}
\end{align*}
\end{columns}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \begin{columns}
    \column{0.5\linewidth}
    \begin{align*}
      \onslide<+->{\D{f}{y}}
      \onslide<+->{& = \D{}{y}\left(\arctan\left(\frac{x}{y}\right)\right)} \\
      \onslide<+->{& = \arctan'\left(\frac{x}{y}\right) \D{}{y}\left(\frac{x}{y}\right)} \\
      \onslide<+->{& = \frac{1}{1+\left(\frac{x}{y}\right)^2} \D{}{y}\left(xy^{-1}\right)} \\
      \onslide<+->{& = \frac{1}{1+\left(\frac{x}{y}\right)^2} \left(-xy^{-2}\right)}
    \end{align*}
    \column{0.5\linewidth}
    \begin{align*}
      \onslide<+->{\D{f}{x}}
      \onslide<+->{& = \frac{1}{1+\frac{x^2}{y^2}}        \left(-xy^{-2}\right)} \\
      \onslide<+->{& = \frac{1}{\;\dfrac{y^2+x^2}{y^2}\;} \left(-xy^{-2}\right)} \\
      \onslide<+->{& = \frac{y^2}{y^2+x^2}                \left(-xy^{-2}\right)} \\
      \onslide<+->{& = \frac{-x}{x^2 + y^2}}
    \end{align*}
  \end{columns}
\end{resolution}
