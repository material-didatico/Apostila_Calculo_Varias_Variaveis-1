\section{Derivadas Parciais}

%------------------------------------------------------------------------------%
\begin{frame}{Derivada Ordinária -- Uma variável}

  Variação da função com relação a uma variável

  \pause
  \vspace{\baselineskip}
  Derivada da função \(f\colon R\subset\R\to\R\) no ponto $a$

  \[
    \frac{df}{dx}(a)
    \;=\; \frac{df}{dx}\evalat{x=a}{}
    =\; \lim_{h\to 0} \frac{\,f(a+h)-f(a)\,}{h}
  \]
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Derivada Parcial}
  Variação da função com relação a \emph{uma} das variáveis

  considerando as demais \emph{fixas}
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Definição -- Derivada Parcial em $x$}
  A derivada parcial, com relação a \emph{$x$},
  da função \(f\colon R\subset\R^2\to\R\),

  \pause
  \(f(x, y)\),
  no ponto \((a, b)\)

  \pause
  \[
    \D{f}{\emph{x}}(a, b)
    = \D{f}{\emph{x}}\evalat{(x, y) = (a, b)}{}
    =\; \lim_{h\to 0} \frac{\,f(\emph{a+h}, b)-f(a, b)\,}{h}
  \]

  \vspace{\baselineskip}
  Desde que o limite exista
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Definição -- Derivada Parcial em $y$}
  A derivada parcial, com relação a \emph{$y$},
  da função \(f\colon R\subset\R^2\to\R\),

  \pause
  \(f(x, y)\),
  no ponto \((a, b)\)

  \pause
  \[
    \D{f}{\emph{y}}(a, b)
    = \D{f}{\emph{y}}\evalat{(x, y) = (a, b)}{}
    =\; \lim_{h\to 0} \frac{\,f(a, \emph{b+h})-f(a, b)\,}{h}
  \]

  \vspace{\baselineskip}
  Desde que o limite exista
\end{frame}

%------------------------------------------------------------------------------%
\begin{frame}{Notações}
  \begin{align*}
     \onslide<+->{& \frac{\partial f}{\partial y}(x, y) &  & \frac{\partial f}{\partial y} \\[11mm]}
     \onslide<+->{& f_y(x, y)                           &  & f_y                           \\[11mm]}
     \onslide<+->{& \partial_y f(x, y)                  &  & \partial_y f}
  \end{align*}
\end{frame}

\begin{frame}{Retas Tangentes}
  Temos duas retas tangentes, uma na direção $x$ e uma na direção $y$

  \pause
  \vspace{\baselineskip}
  Nem sempre o plano gerado por elas é plano tangente da superfície
\end{frame}
