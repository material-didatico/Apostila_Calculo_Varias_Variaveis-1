  \medskip

\end{frame}

\begin{frame}{Derivadas Parciais e Continuidade}
  Se as derivadas parciais de $f$ existem em $(a, b)$

  \pause
  \emph{não podemos dizer nada} sobre a continuidade de $f$ no ponto
\end{frame}
