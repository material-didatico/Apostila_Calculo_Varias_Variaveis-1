%------------------------------------------------------------------------------%
\begin{question}
  Encontre as raízes do polinômio \(p(x) = x^3 -6x^2 + 10x\)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  Queremos resolver a equação
  \[
    x^3 -6x^2 + 10x = 0
  \]
  \pause
  que podemos escrever como
  \[
    x\left(x^2 -6x + 10\right) = 0
  \]
  \pause
  Uma das raízes é \(x=0\), usamos Bhaskara para encontrar as demais
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \[
    \Delta
    \pause
    \;=\; b^2 - 4ac
    \pause
    \;=\; (-6)^2 - 4 \times 1 \times 10
    \pause
    \;=\; 36 - 40
    \pause
    \;=\; -4
  \]

  \pause
  \vspace{0.5\baselineskip}
  \[
    \sqrt{\Delta}
    \pause
    \;=\; \sqrt{-4}
    \pause
    \;=\; i\sqrt{4}
    \pause
    \;=\; 2i
  \]

  \pause
  \vspace{0.5\baselineskip}
  \[
    x
    \pause
    \;=\; \frac{-b\pm\sqrt{\Delta}}{2a}
    \pause
    \;=\; \frac{-(-6)\pm 2i}{2}
    \pause
    \;=\; 3\pm i
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  Portanto as raízes de $p$ são
  \[
    x_1 = 0 \qquad\qquad
    x_2 = 3-i \qquad\qquad
    x_3 = 3+i
  \]
\end{resolution}
