%------------------------------------------------------------------------------%
\begin{question}
  Encontre as raízes cúbicas complexas do número \( z = 1 + i \)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  Escrevendo \(z = 1 + i\) na forma polar

  \pause
  Módulo
  \[
    \rho
    \pause
    = \abs{1+i}
    \pause
    = \sqrt{1^2 + 1^2}
    \pause
    = \sqrt{2}
  \]

  \pause
  Argumento
  \[
    \varphi
    \pause
    = \arg(1+i)
    \pause
    = \ang{45}
    \pause
    = \frac{\pi}{4}
  \]

  \pause
  Forma polar
  \[
    z = \sqrt{2} \left[ \cos\left(\frac{\pi}{4}\right)
                    + i \sin\left(\frac{\pi}{4}\right) \right]
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Raízes cúbicas}
\begin{align*}
  \onslide<+->{u_k}
  \onslide<+->{& = \sqrt[3]{\rho} \left[
  \cos\left(\frac{\varphi + 2k\pi}{3}\right)
  + i \sin\left(\frac{\varphi + 2k\pi}{3}\right)
  \right]
  & k = 0, 1, 2    \\}
  \onslide<+->{& = \sqrt[3]{\sqrt{2}} \left[
  \cos\left(\frac{1}{3}\left(\frac{\pi}{4} + 2k\pi\right)\right)
  + i \sin\left(\frac{1}{3}\left(\frac{\pi}{4} + 2k\pi\right)\right)
  \right] \\}
  \onslide<+->{& = \sqrt[6]{2} \left[
  \cos\left( \frac{\pi + 8k\pi}{12} \right)
  + i \sin\left( \frac{\pi + 8k\pi}{12} \right)
  \right]}
\end{align*}
\end{resolution}

    \right] \\[2mm]}

    \right]}
  \end{align*}
\end{resolution}

\begin{resolution}{Solução}
  \begin{align*}
    \onslide<+->{u_1}
    \onslide<+->{& = \sqrt[6]{2} \left[
    \cos\left( \frac{\pi + 8\times1\pi}{12} \right)
    + i \sin\left( \frac{\pi + 8\times1\pi}{12} \right)
    \right] \\}
    \onslide<+->{& = \sqrt[6]{2} \left[
    \cos\left( \frac{9\pi}{12} \right)
    + i \sin\left( \frac{9\pi}{12} \right)
    \right] \\}
    \onslide<+->{& = \sqrt[6]{2} \left[
    \cos\left( \frac{3\pi}{4} \right)
    + i \sin\left( \frac{3\pi}{4} \right)
    \right] \\}
    \onslide<+->{& = \sqrt[6]{2} \left[
    - \frac{\sqrt{2}}{2}
    + i \frac{\sqrt{2}}{2}
    \right] \\}
    \onslide<+->{& = - \frac{2^{\sfrac{1}{6}}}{2^{\sfrac{1}{2}}}
    + \frac{2^{\sfrac{1}{6}}}{2^{\sfrac{1}{2}}} i  \\}
    \onslide<+->{& = - 2^{-\sfrac{1}{3}}
    + 2^{-\sfrac{1}{3}} i \\[2mm]}
    \onslide<+->{& = - \frac{1}{\sqrt[3]{2}}
    + \frac{i}{\sqrt[3]{2}}}
  \end{align*}
\end{resolution}
