%------------------------------------------------------------------------------%
\begin{question}
  Encontre as raízes cúbicas complexas do número \( z = 27i \)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Forma polar}
  Escrevemos \( z = 27i \) na forma polar
  \pause
  \[
    z
    = 27(0 + i)
    \pause
    = 27 \left[ \cos\left(\frac{\pi}{2}\right) + i \sin\left(\frac{\pi}{2}\right) \right]
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
\begin{align*}
  \onslide<+->{u_k}
  \onslide<+->{& = \sqrt[3]{\rho} \left[
  \cos\left(\frac{\varphi+2k\pi}{3}\right)
  + i\sin\left(\frac{\varphi+2k\pi}{3}\right)
  \right] & k = 0, 1, 2  \\}
  \onslide<+->{& = \sqrt[3]{27} \left[
  \cos\left(\frac{1}{3}\left(\frac{\pi}{2} + 2k\pi\right)\right)
  + i \sin\left(\frac{1}{3}\left(\frac{\pi}{2} + 2k\pi\right)\right)
  \right] \\}
  \onslide<+->{& = 3 \left[
  \cos\left( \frac{\pi + 4k\pi}{6} \right)
  + i \sin\left( \frac{\pi + 4k\pi}{6} \right)
  \right]}
\end{align*}
\end{resolution}
