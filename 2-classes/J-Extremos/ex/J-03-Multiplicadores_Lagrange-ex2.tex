%------------------------------------------------------------------------------%
\begin{question}
  Encontre os valores máximo e mínimo da função
  \[
    f(x, y) = 3x + 4y
  \]
  na circunferência
  \[
    x^2 + y^2 = 1
  \]
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Interpretação do Problema}
  Queremos encontrar os valores extremos da função
  \[
    f(x, y) = 3x + 4y
  \]
  \pause
  sujeito a restrição
  \[
    g(x, y) = x^2 + y^2 - 1 = 0
  \]

  \pause
  Para isso precisamos encontrar os valores $x$, $y$ e $\lambda$ tais que
  \[
    \nabla f = \lambda \nabla g
    \qquad\qquad
    g(x, y) = 0
  \]
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Gradientes}
\begin{columns}
\column{0.4\linewidth}
  Gradiente de $f$
  \[
    \pause
    \D{f}{x}
    \pause
    = \D{}{x}\left(3x + 4y\right)
    \pause
    = 3
  \]
  \[
    \pause
    \D{f}{y}
    \pause
    = \D{}{y}\left(3x + 4y\right)
    \pause
    = 4
  \]
  \pause
  \[
    {\color<17>{blue}\nabla f = \vetor{3}{4}}
  \]
\column{0.6\linewidth}
  \pause
  Gradiente de $g$
  \[
    \pause
    \D{g}{x}
    \pause
    = \D{}{x}\left(x^2 + y^2 - 1\right)
    \pause
    = 2x
  \]
  \[
    \pause
    \D{g}{y}
    \pause
    = \D{}{y}\left(x^2 + y^2 - 1\right)
    \pause
    = 2y
  \]
  \pause
  \[
    {\color<17>{blue}\nabla g = \vetor{2x}{2y}}
  \]
\end{columns}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Sistema}
\begin{columns}[t]
\column{0.4\linewidth}
  A equação
  \[
    \nabla f = \lambda \nabla g
  \]
  \pause
  se torna
  \[
    \vetor{3}{4} = \lambda \vetor{2x}{2y}
  \]
  {\color<7>{blue}%
    \pause
    \[
    \begin{cases}
      2x\lambda = 3 \\
      2y\lambda = 4
    \end{cases}
  \]}
\column{0.4\linewidth}
  \pause
  A equação
  \[
    g(x, y) = 0
  \]
  \pause
  se torna
  \[
    x^2 + y^2 - 1 = 0
  \]
  {\color<7>{blue}%
    \pause
  \[
    x^2 + y^2 = 1
  \]}
\end{columns}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
\begin{columns}
\column{0.4\linewidth}
  Precisamos resolver o sistema
  \[
  \begin{cases}
    2x\lambda = 3 \\
    2y\lambda = 4 \\
    x^2 + y^2 = 1
  \end{cases}
  \]
  \pause
  Notando que \(\lambda\neq 0\)
  \[
    \begin{cases}
      x = \dfrac{3}{2\lambda} \\[2mm]
      y = \dfrac{2}{ \lambda} \\[2mm]
      x^2 + y^2 = 1
    \end{cases}
    \pause
  \]
\column{0.4\linewidth}
  \begin{align*}
    \onslide<+->{x^2 + y^2 & = 1                         \\}
    \onslide<+->{\left(\frac{3}{2\lambda}\right)^2 +
                 \left(\frac{2}{\lambda}\right)^2 & = 1  \\}
    \onslide<+->{\frac{9}{4\lambda^2} +
                 \frac{4}{\lambda^2} & = 1               \\}
    \onslide<+->{\frac{9+16}{4\lambda^2} & = 1           \\}
    \onslide<+->{4\lambda^2 & = 25                       \\}
    \onslide<+->{\lambda & = \pm \frac{5}{2}}
  \end{align*}
\end{columns}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
\begin{columns}[t]
\column{0.4\linewidth}
  Quando \(\lambda=\frac{5}{2}\)
  \[
    \pause
    x
    = \frac{3}{2\lambda}
    \pause
    = \frac{3}{2\sfrac{5}{2}}
    \pause
    = \frac{3}{5}
  \]
  \[
    \pause
        y
    = \frac{2}{\lambda}
    \pause
    = \frac{2}{\sfrac{5}{2}}
    \pause
    = \frac{4}{5}
  \]
\column{0.4\linewidth}
  \pause
  Quando \(\lambda=-\sfrac{5}{2}\)
  \[
    \pause
    x
    = \frac{3}{2\lambda}
    \pause
    = -\frac{3}{2\sfrac{5}{2}}
    \pause
    = -\frac{3}{5}
  \]
  \[
    \pause
    y
    = \frac{2}{\lambda}
    \pause
    = -\frac{2}{\sfrac{5}{2}}
    \pause
    = -\frac{4}{5}
  \]
\end{columns}
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \onslide<+->{Avaliando a função \(f(x, y)=3x + 4y\) nos pontos}
  \begin{align*}
    \onslide<+->{f\left(           -\frac{3}{5},            -\frac{4}{5}\right) &= }
    \onslide<+->{3\left(-\frac{3}{5}\right) + 4\left(-\frac{4}{5}\right) = }
    \onslide<+->{\frac{-9-16}{5} = }
    \onslide<+->{-5 \\[5mm]}
    \onslide<+->{f\left(\hphantom{-}\frac{3}{5}, \hphantom{-}\frac{4}{5}\right) &= }
    \onslide<+->{3\frac{3}{5} + 4\frac{4}{5} = }
    \onslide<+->{\frac{9+16}{5} = }
    \onslide<+->{5}
  \end{align*}

  \onslide<+->{O valor máximo de $f$ é $\hpm 5$ e ocorre no ponto
  \(\left(\frac{3}{5}, \frac{4}{5}\right)\)}

  \vspace{0.3\baselineskip}
  \onslide<+->{O valor mínimo de $f$ é $-5$ e ocorre no ponto
  \(\left(-\frac{3}{5}, -\frac{4}{5}\right)\)}
\end{resolution}
