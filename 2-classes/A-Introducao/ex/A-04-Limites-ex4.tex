%------------------------------------------------------------------------------%
\begin{question}
  Avalie
  \(
    \lim_{x \to 0^+} x\ln(x)
  \)
\end{question}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  Não podemos substituir diretamente, mas sabemos que
  \[
    \lim_{x \to 0^+} x = 0
  \]
  \[
    \lim_{x \to 0^+} \ln x = -\infty
  \]
  portanto temos uma indeterminação \(0(-\infty)\)
  e não podemos dizer nada sobre o limite
\end{resolution}

%------------------------------------------------------------------------------%
\begin{resolution}{Solução}
  \begin{align*}
    \lim_{x \to 0^+} x\ln(x)
    & = \lim_{x \to 0^+} \frac{\ln(x)}{\sfrac{1}{x}}  & & \text{indeterminação } \frac{\infty}{\infty}\\
    & = \lim_{x \to 0^+} \frac{\left(\ln(x)\right)'}{\left(x^{-1}\right)'} & & \text{regra de L'Hôpital} \\
    & = \lim_{x \to 0^+} \frac{x^{-1}}{-x^{-2}} \\
    & = \lim_{x \to 0^+} \frac{x^2}{-x} \\
    & = \lim_{x \to 0^+} -x \;=\; 0
  \end{align*}
\end{resolution}
